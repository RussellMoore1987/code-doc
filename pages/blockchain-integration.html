const Web3 = require('web3');
const { EventEmitter } = require('events');

class Web3Manager extends EventEmitter {
  constructor(config = {}) {
    super();
    
    this.config = {
      defaultNetwork: config.defaultNetwork || 'localhost',
      networks: {
        localhost: {
          rpc: 'http://localhost:8545',
          chainId: 1337,
          name: 'Local Development'
        },
        mainnet: {
          rpc: process.env.MAINNET_RPC_URL,
          chainId: 1,
          name: 'Ethereum Mainnet'
        },
        sepolia: {
          rpc: process.env.SEPOLIA_RPC_URL,
          chainId: 11155111,
          name: 'Sepolia Testnet'
        },
        polygon: {
          rpc: process.env.POLYGON_RPC_URL,
          chainId: 137,
          name: 'Polygon Mainnet'
        },
        ...config.networks
      },
      gasSettings: {
        gasPrice: config.gasPrice || 'auto',
        gasLimit: config.gasLimit || 6000000,
        maxFeePerGas: config.maxFeePerGas,
        maxPriorityFeePerGas: config.maxPriorityFeePerGas
      },
      retrySettings: {
        maxRetries: config.maxRetries || 3,
        retryDelay: config.retryDelay || 1000
      }
    };
    
    this.web3Instance = null;
    this.currentNetwork = null;
    this.accounts = [];
    this.contracts = new Map();
    this.transactionQueue = [];
    this.isConnected = false;
  }

  // Initialize Web3 connection
  async connect(networkName = this.config.defaultNetwork) {
    const network = this.config.networks[networkName];
    if (!network) {
      throw new Error(`Network ${networkName} not configured`);
    }

    try {
      this.web3Instance = new Web3(new Web3.providers.HttpProvider(network.rpc));
      this.currentNetwork = { name: networkName, ...network };
      
      // Test connection
      await this.web3Instance.eth.getBlockNumber();
      this.isConnected = true;
      
      // Get accounts if available (development)
      try {
        this.accounts = await this.web3Instance.eth.getAccounts();
      } catch (error) {
        console.log('No accounts available (likely mainnet/testnet)');
      }

      console.log(`Connected to ${network.name} (Chain ID: ${network.chainId})`);
      this.emit('connected', { network: this.currentNetwork, accounts: this.accounts });
      
      return this.web3Instance;
    } catch (error) {
      this.isConnected = false;
      console.error(`Failed to connect to ${networkName}:`, error.message);
      throw error;
    }
  }

  // Get current gas prices
  async getGasInfo() {
    if (!this.isConnected) {
      throw new Error('Not connected to any network');
    }

    try {
      const gasPrice = await this.web3Instance.eth.getGasPrice();
      const block = await this.web3Instance.eth.getBlock('latest');
      
      const gasInfo = {
        gasPrice: gasPrice,
        gasPriceGwei: this.web3Instance.utils.fromWei(gasPrice, 'gwei'),
        baseFeePerGas: block.baseFeePerGas,
        network: this.currentNetwork.name
      };

      // Estimate gas fees for different priorities (EIP-1559)
      if (block.baseFeePerGas) {
        const baseFee = BigInt(block.baseFeePerGas);
        const priorityFees = {
          slow: baseFee + BigInt(this.web3Instance.utils.toWei('1', 'gwei')),
          standard: baseFee + BigInt(this.web3Instance.utils.toWei('2', 'gwei')),
          fast: baseFee + BigInt(this.web3Instance.utils.toWei('3', 'gwei'))
        };

        gasInfo.estimatedFees = {
          slow: {
            maxFeePerGas: priorityFees.slow.toString(),
            maxPriorityFeePerGas: this.web3Instance.utils.toWei('1', 'gwei')
          },
          standard: {
            maxFeePerGas: priorityFees.standard.toString(),
            maxPriorityFeePerGas: this.web3Instance.utils.toWei('2', 'gwei')
          },
          fast: {
            maxFeePerGas: priorityFees.fast.toString(),
            maxPriorityFeePerGas: this.web3Instance.utils.toWei('3', 'gwei')
          }
        };
      }

      return gasInfo;
    } catch (error) {
      console.error('Failed to get gas info:', error);
      throw error;
    }
  }

  // Load smart contract
  loadContract(abi, address, name) {
    if (!this.isConnected) {
      throw new Error('Not connected to any network');
    }

    try {
      const contract = new this.web3Instance.eth.Contract(abi, address);
      this.contracts.set(name, {
        instance: contract,
        address,
        abi,
        name
      });

      console.log(`Contract ${name} loaded at ${address}`);
      return contract;
    } catch (error) {
      console.error(`Failed to load contract ${name}:`, error);
      throw error;
    }
  }

  // Get contract instance
  getContract(name) {
    const contract = this.contracts.get(name);
    return contract ? contract.instance : null;
  }

  // Send transaction with retry logic
  async sendTransaction(transactionConfig, options = {}) {
    if (!this.isConnected) {
      throw new Error('Not connected to any network');
    }

    const config = {
      ...transactionConfig,
      ...await this.prepareGasSettings(options.gasSettings)
    };

    let attempt = 0;
    let lastError;

    while (attempt < this.config.retrySettings.maxRetries) {
      try {
        const receipt = await this.web3Instance.eth.sendTransaction(config);
        
        this.emit('transactionSent', {
          hash: receipt.transactionHash,
          blockNumber: receipt.blockNumber,
          gasUsed: receipt.gasUsed
        });

        return receipt;
      } catch (error) {
        lastError = error;
        attempt++;
        
        if (attempt < this.config.retrySettings.maxRetries) {
          console.log(`Transaction failed, retrying (${attempt}/${this.config.retrySettings.maxRetries})`);
          await this.delay(this.config.retrySettings.retryDelay * attempt);
        }
      }
    }

    throw new Error(`Transaction failed after ${this.config.retrySettings.maxRetries} attempts: ${lastError.message}`);
  }

  // Call contract method
  async callContractMethod(contractName, methodName, params = [], options = {}) {
    const contract = this.getContract(contractName);
    if (!contract) {
      throw new Error(`Contract ${contractName} not found`);
    }

    try {
      if (options.send) {
        // Send transaction (state-changing)
        const gasEstimate = await contract.methods[methodName](...params).estimateGas({
          from: options.from
        });

        const gasSettings = await this.prepareGasSettings({
          ...options.gasSettings,
          gas: Math.floor(gasEstimate * 1.2) // Add 20% buffer
        });

        const result = await contract.methods[methodName](...params).send({
          from: options.from,
          ...gasSettings
        });

        this.emit('contractTransaction', {
          contract: contractName,
          method: methodName,
          hash: result.transactionHash,
          blockNumber: result.blockNumber
        });

        return result;
      } else {
        // Call method (read-only)
        return await contract.methods[methodName](...params).call(options.callOptions || {});
      }
    } catch (error) {
      console.error(`Contract call failed (${contractName}.${methodName}):`, error);
      throw error;
    }
  }

  // Prepare gas settings
  async prepareGasSettings(customGas = {}) {
    const gasInfo = await this.getGasInfo();
    
    const settings = {
      gas: customGas.gas || this.config.gasSettings.gasLimit,
      gasPrice: customGas.gasPrice || this.config.gasSettings.gasPrice
    };

    // Handle 'auto' gas price
    if (settings.gasPrice === 'auto') {
      if (gasInfo.estimatedFees) {
        // Use EIP-1559
        settings.maxFeePerGas = customGas.maxFeePerGas || gasInfo.estimatedFees.standard.maxFeePerGas;
        settings.maxPriorityFeePerGas = customGas.maxPriorityFeePerGas || gasInfo.estimatedFees.standard.maxPriorityFeePerGas;
        delete settings.gasPrice;
      } else {
        // Use legacy gas pricing
        settings.gasPrice = gasInfo.gasPrice;
      }
    }

    return settings;
  }

  // Listen to contract events
  subscribeToEvent(contractName, eventName, options = {}, callback) {
    const contract = this.getContract(contractName);
    if (!contract) {
      throw new Error(`Contract ${contractName} not found`);
    }

    const subscription = contract.events[eventName](options);
    
    subscription.on('data', (event) => {
      console.log(`Event received: ${contractName}.${eventName}`);
      callback(null, event);
    });

    subscription.on('error', (error) => {
      console.error(`Event error (${contractName}.${eventName}):`, error);
      callback(error, null);
    });

    return subscription;
  }

  // Get account balance
  async getBalance(address, unit = 'ether') {
    if (!this.isConnected) {
      throw new Error('Not connected to any network');
    }

    try {
      const balance = await this.web3Instance.eth.getBalance(address);
      return this.web3Instance.utils.fromWei(balance, unit);
    } catch (error) {
      console.error(`Failed to get balance for ${address}:`, error);
      throw error;
    }
  }

  // Get transaction receipt
  async getTransactionReceipt(hash) {
    if (!this.isConnected) {
      throw new Error('Not connected to any network');
    }

    return await this.web3Instance.eth.getTransactionReceipt(hash);
  }

  // Wait for transaction confirmation
  async waitForTransaction(hash, confirmations = 1, timeout = 300000) {
    return new Promise((resolve, reject) => {
      let confirmedBlocks = 0;
      let transactionReceipt = null;
      
      const timeoutId = setTimeout(() => {
        reject(new Error(`Transaction timeout: ${hash}`));
      }, timeout);

      const checkTransaction = async () => {
        try {
          const receipt = await this.getTransactionReceipt(hash);
          
          if (receipt) {
            if (!transactionReceipt) {
              transactionReceipt = receipt;
            }
            
            const currentBlock = await this.web3Instance.eth.getBlockNumber();
            confirmedBlocks = currentBlock - receipt.blockNumber;
            
            if (confirmedBlocks >= confirmations) {
              clearTimeout(timeoutId);
              resolve(receipt);
              return;
            }
          }
          
          setTimeout(checkTransaction, 5000);
        } catch (error) {
          clearTimeout(timeoutId);
          reject(error);
        }
      };

      checkTransaction();
    });
  }

  // Utility method to create delay
  delay(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  // Get network info
  getNetworkInfo() {
    return this.currentNetwork;
  }

  // Disconnect
  disconnect() {
    this.isConnected = false;
    this.web3Instance = null;
    this.currentNetwork = null;
    this.accounts = [];
    this.contracts.clear();
    
    this.emit('disconnected');
    console.log('Disconnected from blockchain');
  }
}

const solc = require('solc');
const fs = require('fs').promises;
const path = require('path');

class SmartContractManager {
  constructor(web3Manager) {
    this.web3Manager = web3Manager;
    this.compiledContracts = new Map();
    this.deployedContracts = new Map();
    this.contractSources = new Map();
  }

  // Compile Solidity contract
  async compileContract(contractPath, contractName) {
    try {
      const source = await fs.readFile(contractPath, 'utf8');
      this.contractSources.set(contractName, source);

      const input = {
        language: 'Solidity',
        sources: {
          [contractName]: {
            content: source
          }
        },
        settings: {
          outputSelection: {
            '*': {
              '*': ['*']
            }
          }
        }
      };

      const output = JSON.parse(solc.compile(JSON.stringify(input)));

      if (output.errors) {
        const errors = output.errors.filter(error => error.severity === 'error');
        if (errors.length > 0) {
          throw new Error(`Compilation errors: ${errors.map(e => e.message).join(', ')}`);
        }
      }

      const contract = output.contracts[contractName][contractName];
      const compiled = {
        name: contractName,
        abi: contract.abi,
        bytecode: contract.evm.bytecode.object,
        gasEstimate: contract.evm.gasEstimates.creation.totalCost,
        source
      };

      this.compiledContracts.set(contractName, compiled);
      console.log(`Contract ${contractName} compiled successfully`);
      
      return compiled;
    } catch (error) {
      console.error(`Failed to compile contract ${contractName}:`, error);
      throw error;
    }
  }

  // Deploy contract
  async deployContract(contractName, constructorArgs = [], options = {}) {
    const compiled = this.compiledContracts.get(contractName);
    if (!compiled) {
      throw new Error(`Contract ${contractName} not compiled`);
    }

    try {
      const web3 = this.web3Manager.web3Instance;
      const contract = new web3.eth.Contract(compiled.abi);
      
      // Estimate gas for deployment
      const estimatedGas = await contract.deploy({
        data: '0x' + compiled.bytecode,
        arguments: constructorArgs
      }).estimateGas();

      const gasSettings = await this.web3Manager.prepareGasSettings({
        gas: Math.floor(estimatedGas * 1.2), // Add 20% buffer
        ...options.gasSettings
      });

      console.log(`Deploying ${contractName} with gas limit: ${gasSettings.gas}`);

      const deployedContract = await contract.deploy({
        data: '0x' + compiled.bytecode,
        arguments: constructorArgs
      }).send({
        from: options.from,
        ...gasSettings
      });

      const deployment = {
        name: contractName,
        address: deployedContract.options.address,
        transactionHash: deployedContract.transactionHash,
        blockNumber: deployedContract.blockNumber,
        abi: compiled.abi,
        deployedAt: new Date().toISOString(),
        constructorArgs,
        gasUsed: deployedContract.gasUsed
      };

      this.deployedContracts.set(contractName, deployment);
      
      // Load contract into Web3Manager
      this.web3Manager.loadContract(compiled.abi, deployment.address, contractName);

      console.log(`Contract ${contractName} deployed at ${deployment.address}`);
      return deployment;
    } catch (error) {
      console.error(`Failed to deploy contract ${contractName}:`, error);
      throw error;
    }
  }

  // Verify contract on blockchain explorer
  async verifyContract(contractName, explorerApiKey, explorerUrl) {
    const deployment = this.deployedContracts.get(contractName);
    const source = this.contractSources.get(contractName);
    
    if (!deployment || !source) {
      throw new Error(`Contract ${contractName} not found or not deployed`);
    }

    // This would integrate with blockchain explorer APIs like Etherscan
    console.log(`Verifying contract ${contractName} at ${deployment.address}`);
    
    // Simulated verification
    return {
      contractName,
      address: deployment.address,
      verified: true,
      explorerUrl: `${explorerUrl}/address/${deployment.address}#code`
    };
  }

  // Upgrade contract (proxy pattern)
  async upgradeContract(proxyName, newImplementationName, options = {}) {
    const proxy = this.deployedContracts.get(proxyName);
    const newImplementation = this.deployedContracts.get(newImplementationName);
    
    if (!proxy || !newImplementation) {
      throw new Error('Proxy or implementation contract not found');
    }

    try {
      // Call upgrade function on proxy
      const result = await this.web3Manager.callContractMethod(
        proxyName,
        'upgrade',
        [newImplementation.address],
        {
          send: true,
          from: options.from,
          gasSettings: options.gasSettings
        }
      );

      console.log(`Contract ${proxyName} upgraded to ${newImplementation.address}`);
      return result;
    } catch (error) {
      console.error(`Failed to upgrade contract ${proxyName}:`, error);
      throw error;
    }
  }

  // Generate contract documentation
  generateContractDocs(contractName) {
    const compiled = this.compiledContracts.get(contractName);
    const deployment = this.deployedContracts.get(contractName);
    
    if (!compiled) {
      throw new Error(`Contract ${contractName} not compiled`);
    }

    const functions = compiled.abi.filter(item => item.type === 'function');
    const events = compiled.abi.filter(item => item.type === 'event');
    
    const docs = {
      name: contractName,
      address: deployment?.address,
      functions: functions.map(func => ({
        name: func.name,
        inputs: func.inputs.map(input => ({
          name: input.name,
          type: input.type
        })),
        outputs: func.outputs?.map(output => ({
          name: output.name,
          type: output.type
        })),
        stateMutability: func.stateMutability,
        payable: func.payable
      })),
      events: events.map(event => ({
        name: event.name,
        inputs: event.inputs.map(input => ({
          name: input.name,
          type: input.type,
          indexed: input.indexed
        }))
      }))
    };

    return docs;
  }

  // Get deployment info
  getDeploymentInfo(contractName) {
    return this.deployedContracts.get(contractName);
  }

  // List all deployed contracts
  listDeployedContracts() {
    return Array.from(this.deployedContracts.values());
  }

  // Save deployment to file
  async saveDeployment(contractName, filePath) {
    const deployment = this.deployedContracts.get(contractName);
    if (!deployment) {
      throw new Error(`Contract ${contractName} not deployed`);
    }

    try {
      await fs.writeFile(filePath, JSON.stringify(deployment, null, 2));
      console.log(`Deployment saved to ${filePath}`);
    } catch (error) {
      console.error(`Failed to save deployment:`, error);
      throw error;
    }
  }

  // Load deployment from file
  async loadDeployment(filePath, contractName) {
    try {
      const data = await fs.readFile(filePath, 'utf8');
      const deployment = JSON.parse(data);
      
      this.deployedContracts.set(contractName, deployment);
      
      // Load contract into Web3Manager
      this.web3Manager.loadContract(deployment.abi, deployment.address, contractName);
      
      console.log(`Deployment loaded for ${contractName} at ${deployment.address}`);
      return deployment;
    } catch (error) {
      console.error(`Failed to load deployment:`, error);
      throw error;
    }
  }
}

class CryptocurrencyPaymentProcessor {
  constructor(web3Manager) {
    this.web3Manager = web3Manager;
    this.paymentHistory = new Map();
    this.pendingPayments = new Map();
    this.webhooks = [];
    this.supportedTokens = new Map();
    
    this.setupDefaultTokens();
  }

  setupDefaultTokens() {
    // Standard ERC-20 ABI for token transfers
    const erc20ABI = [
      {
        "constant": true,
        "inputs": [{"name": "_owner", "type": "address"}],
        "name": "balanceOf",
        "outputs": [{"name": "balance", "type": "uint256"}],
        "type": "function"
      },
      {
        "constant": false,
        "inputs": [
          {"name": "_to", "type": "address"},
          {"name": "_value", "type": "uint256"}
        ],
        "name": "transfer",
        "outputs": [{"name": "", "type": "bool"}],
        "type": "function"
      },
      {
        "constant": true,
        "inputs": [],
        "name": "decimals",
        "outputs": [{"name": "", "type": "uint8"}],
        "type": "function"
      },
      {
        "anonymous": false,
        "inputs": [
          {"indexed": true, "name": "from", "type": "address"},
          {"indexed": true, "name": "to", "type": "address"},
          {"indexed": false, "name": "value", "type": "uint256"}
        ],
        "name": "Transfer",
        "type": "event"
      }
    ];

    // Add common tokens (these would be configured per network)
    this.supportedTokens.set('ETH', {
      symbol: 'ETH',
      name: 'Ethereum',
      decimals: 18,
      isNative: true
    });

    this.supportedTokens.set('USDC', {
      symbol: 'USDC',
      name: 'USD Coin',
      decimals: 6,
      isNative: false,
      address: '0xA0b86a33E6441B4e33d4d7a70a3E1Ab3e8F8E7D7', // Example address
      abi: erc20ABI
    });

    this.supportedTokens.set('DAI', {
      symbol: 'DAI',
      name: 'Dai Stablecoin',
      decimals: 18,
      isNative: false,
      address: '0x6B175474E89094C44Da98b954EedeAC495271d0F', // Example address
      abi: erc20ABI
    });
  }

  // Create payment request
  async createPaymentRequest(config) {
    const paymentId = this.generatePaymentId();
    const payment = {
      id: paymentId,
      amount: config.amount,
      currency: config.currency || 'ETH',
      recipient: config.recipient,
      description: config.description || '',
      metadata: config.metadata || {},
      status: 'pending',
      created: new Date().toISOString(),
      expiresAt: config.expiresAt || new Date(Date.now() + 3600000).toISOString(), // 1 hour default
      confirmationsRequired: config.confirmationsRequired || 3,
      confirmations: 0,
      transactionHash: null,
      blockNumber: null
    };

    this.pendingPayments.set(paymentId, payment);

    console.log(`Payment request created: ${paymentId} for ${config.amount} ${config.currency}`);
    return payment;
  }

  // Process native ETH payment
  async processEthPayment(paymentId, fromAddress, privateKey, options = {}) {
    const payment = this.pendingPayments.get(paymentId);
    if (!payment) {
      throw new Error('Payment request not found');
    }

    if (payment.currency !== 'ETH') {
      throw new Error('Invalid currency for ETH payment');
    }

    try {
      const web3 = this.web3Manager.web3Instance;
      const amountWei = web3.utils.toWei(payment.amount.toString(), 'ether');
      
      // Check sender balance
      const balance = await web3.eth.getBalance(fromAddress);
      if (BigInt(balance) < BigInt(amountWei)) {
        throw new Error('Insufficient balance');
      }

      // Prepare transaction
      const gasSettings = await this.web3Manager.prepareGasSettings(options.gasSettings);
      const nonce = await web3.eth.getTransactionCount(fromAddress, 'pending');

      const transaction = {
        from: fromAddress,
        to: payment.recipient,
        value: amountWei,
        nonce: nonce,
        ...gasSettings
      };

      // Sign and send transaction
      const signedTx = await web3.eth.accounts.signTransaction(transaction, privateKey);
      const receipt = await web3.eth.sendSignedTransaction(signedTx.rawTransaction);

      // Update payment
      payment.status = 'sent';
      payment.transactionHash = receipt.transactionHash;
      payment.blockNumber = receipt.blockNumber;
      payment.gasUsed = receipt.gasUsed;
      payment.sentAt = new Date().toISOString();

      this.paymentHistory.set(paymentId, payment);
      this.pendingPayments.delete(paymentId);

      // Start monitoring for confirmations
      this.monitorPaymentConfirmations(paymentId);

      console.log(`ETH payment sent: ${receipt.transactionHash}`);
      return { payment, receipt };
    } catch (error) {
      payment.status = 'failed';
      payment.error = error.message;
      payment.failedAt = new Date().toISOString();
      
      console.error(`ETH payment failed:`, error);
      throw error;
    }
  }

  // Process ERC-20 token payment
  async processTokenPayment(paymentId, fromAddress, privateKey, options = {}) {
    const payment = this.pendingPayments.get(paymentId);
    if (!payment) {
      throw new Error('Payment request not found');
    }

    const tokenInfo = this.supportedTokens.get(payment.currency);
    if (!tokenInfo || tokenInfo.isNative) {
      throw new Error(`Unsupported token: ${payment.currency}`);
    }

    try {
      // Load token contract
      this.web3Manager.loadContract(tokenInfo.abi, tokenInfo.address, payment.currency);
      
      const web3 = this.web3Manager.web3Instance;
      const tokenAmount = this.parseTokenAmount(payment.amount, tokenInfo.decimals);
      
      // Check token balance
      const balance = await this.web3Manager.callContractMethod(
        payment.currency,
        'balanceOf',
        [fromAddress]
      );

      if (BigInt(balance) < BigInt(tokenAmount)) {
        throw new Error(`Insufficient ${payment.currency} balance`);
      }

      // Estimate gas for token transfer
      const contract = this.web3Manager.getContract(payment.currency);
      const gasEstimate = await contract.methods.transfer(payment.recipient, tokenAmount)
        .estimateGas({ from: fromAddress });

      const gasSettings = await this.web3Manager.prepareGasSettings({
        gas: Math.floor(gasEstimate * 1.2),
        ...options.gasSettings
      });

      const nonce = await web3.eth.getTransactionCount(fromAddress, 'pending');

      // Prepare token transfer transaction
      const transferData = contract.methods.transfer(payment.recipient, tokenAmount).encodeABI();
      
      const transaction = {
        from: fromAddress,
        to: tokenInfo.address,
        data: transferData,
        nonce: nonce,
        ...gasSettings
      };

      // Sign and send transaction
      const signedTx = await web3.eth.accounts.signTransaction(transaction, privateKey);
      const receipt = await web3.eth.sendSignedTransaction(signedTx.rawTransaction);

      // Update payment
      payment.status = 'sent';
      payment.transactionHash = receipt.transactionHash;
      payment.blockNumber = receipt.blockNumber;
      payment.gasUsed = receipt.gasUsed;
      payment.sentAt = new Date().toISOString();

      this.paymentHistory.set(paymentId, payment);
      this.pendingPayments.delete(paymentId);

      // Start monitoring for confirmations
      this.monitorPaymentConfirmations(paymentId);

      console.log(`${payment.currency} payment sent: ${receipt.transactionHash}`);
      return { payment, receipt };
    } catch (error) {
      payment.status = 'failed';
      payment.error = error.message;
      payment.failedAt = new Date().toISOString();
      
      console.error(`Token payment failed:`, error);
      throw error;
    }
  }

  // Monitor payment confirmations
  async monitorPaymentConfirmations(paymentId) {
    const payment = this.paymentHistory.get(paymentId);
    if (!payment || !payment.transactionHash) return;

    const checkConfirmations = async () => {
      try {
        const receipt = await this.web3Manager.getTransactionReceipt(payment.transactionHash);
        if (!receipt) return;

        const currentBlock = await this.web3Manager.web3Instance.eth.getBlockNumber();
        const confirmations = currentBlock - receipt.blockNumber;

        payment.confirmations = confirmations;

        if (confirmations >= payment.confirmationsRequired) {
          payment.status = 'confirmed';
          payment.confirmedAt = new Date().toISOString();
          
          console.log(`Payment confirmed: ${paymentId} (${confirmations} confirmations)`);
          
          // Trigger webhooks
          await this.triggerWebhooks('payment_confirmed', payment);
          return;
        }

        // Continue monitoring
        setTimeout(checkConfirmations, 15000); // Check every 15 seconds
      } catch (error) {
        console.error(`Error monitoring confirmations for ${paymentId}:`, error);
      }
    };

    checkConfirmations();
  }

  // Add webhook
  addWebhook(url, events = ['payment_confirmed']) {
    this.webhooks.push({ url, events });
  }

  // Trigger webhooks
  async triggerWebhooks(event, data) {
    const relevantWebhooks = this.webhooks.filter(hook => hook.events.includes(event));
    
    for (const webhook of relevantWebhooks) {
      try {
        await fetch(webhook.url, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ event, data, timestamp: new Date().toISOString() })
        });
        
        console.log(`Webhook triggered: ${webhook.url}`);
      } catch (error) {
        console.error(`Webhook failed: ${webhook.url}`, error);
      }
    }
  }

  // Get payment status
  getPaymentStatus(paymentId) {
    return this.paymentHistory.get(paymentId) || this.pendingPayments.get(paymentId);
  }

  // List payments
  listPayments(filters = {}) {
    let payments = Array.from(this.paymentHistory.values());
    
    if (filters.status) {
      payments = payments.filter(p => p.status === filters.status);
    }
    
    if (filters.currency) {
      payments = payments.filter(p => p.currency === filters.currency);
    }
    
    if (filters.startDate) {
      payments = payments.filter(p => new Date(p.created) >= new Date(filters.startDate));
    }
    
    if (filters.endDate) {
      payments = payments.filter(p => new Date(p.created) <= new Date(filters.endDate));
    }

    return payments.sort((a, b) => new Date(b.created) - new Date(a.created));
  }

  // Parse token amount with decimals
  parseTokenAmount(amount, decimals) {
    const web3 = this.web3Manager.web3Instance;
    return web3.utils.toWei(amount.toString(), 'ether').slice(0, -18 + decimals);
  }

  // Format token amount for display
  formatTokenAmount(amount, decimals) {
    const divisor = BigInt(10) ** BigInt(decimals);
    const quotient = BigInt(amount) / divisor;
    const remainder = BigInt(amount) % divisor;
    
    return parseFloat(`${quotient}.${remainder.toString().padStart(decimals, '0')}`);
  }

  // Generate payment ID
  generatePaymentId() {
    return 'pay_' + Math.random().toString(36).substr(2, 9) + Date.now().toString(36);
  }

  // Get supported tokens
  getSupportedTokens() {
    return Array.from(this.supportedTokens.values());
  }

  // Add custom token
  addToken(symbol, config) {
    this.supportedTokens.set(symbol, {
      symbol,
      name: config.name,
      decimals: config.decimals,
      address: config.address,
      abi: config.abi || this.supportedTokens.get('USDC').abi,
      isNative: false
    });
    
    console.log(`Token ${symbol} added`);
  }
}
