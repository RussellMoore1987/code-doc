Request Validation

Implement comprehensive request validation to ensure data integrity, security, and API reliability. This guide covers validation schemas, sanitization techniques, error handling, and advanced validation patterns.

🛡️ Security Layer

Request validation is your first line of defense against malicious input, data corruption, and API abuse. Proper validation prevents injection attacks, ensures data consistency, and improves overall application reliability.

Define structured validation schemas that are maintainable and reusable across endpoints:

Validation Middleware

Express Middleware

// Validation middleware factory
const validate = (schema, options = {}) => {
  return async (req, res, next) => {
    try {
      const {
        source = 'body',
        allowUnknown = false,
        stripUnknown = true,
        abortEarly = false
      } = options;

      const dataToValidate = req[source];
      
      const { error, value } = await schema.validateAsync(dataToValidate, {
        allowUnknown,
        stripUnknown,
        abortEarly,
        context: {
          req,
          user: req.user,
          ip: req.ip
        }
      });

      if (error) {
        return res.status(400).json({
          error: 'Validation failed',
          details: formatValidationErrors(error),
          timestamp: new Date().toISOString()
        });
      }

      // Replace the request data with validated/sanitized data
      req[source] = value;
      next();

    } catch (err) {
      console.error('Validation middleware error:', err);
      res.status(500).json({
        error: 'Internal validation error'
      });
    }
  };
};

// Comprehensive validation for multiple sources
const validateRequest = (schemas) => {
  return async (req, res, next) => {
    const errors = {};
    
    for (const [source, schema] of Object.entries(schemas)) {
      if (req[source]) {
        try {
          const { error, value } = await schema.validateAsync(req[source], {
            stripUnknown: true,
            allowUnknown: false
          });
          
          if (error) {
            errors[source] = formatValidationErrors(error);
          } else {
            req[source] = value;
          }
        } catch (err) {
          errors[source] = [`Invalid ${source} data`];
        }
      }
    }
    
    if (Object.keys(errors).length > 0) {
      return res.status(400).json({
        error: 'Request validation failed',
        details: errors
      });
    }
    
    next();
  };
};

Error Formatting

function formatValidationErrors(joiError) {
  return joiError.details.map(detail => ({
    field: detail.path.join('.'),
    message: detail.message,
    value: detail.context?.value,
    type: detail.type
  }));
}

// Advanced error handler with context
class ValidationErrorHandler {
  static format(error, req) {
    const baseError = {
      error: 'Validation failed',
      timestamp: new Date().toISOString(),
      path: req.path,
      method: req.method
    };

    if (error.isJoi) {
      return {
        ...baseError,
        type: 'validation',
        details: error.details.map(detail => ({
          field: detail.path.join('.'),
          message: detail.message,
          rejectedValue: detail.context?.value,
          constraint: detail.type
        }))
      };
    }

    if (error.code === 'LIMIT_FILE_SIZE') {
      return {
        ...baseError,
        type: 'file_upload',
        message: 'File too large',
        limit: error.limit
      };
    }

    return {
      ...baseError,
      type: 'unknown',
      message: 'An error occurred during validation'
    };
  }

  static getHttpStatus(error) {
    if (error.isJoi) {
      return 400; // Bad Request
    }
    
    if (error.code === 'LIMIT_FILE_SIZE') {
      return 413; // Payload Too Large
    }
    
    return 500; // Internal Server Error
  }
}

Data Sanitization

Input Sanitization

const validator = require('validator');
const xss = require('xss');

class DataSanitizer {
  static sanitizeString(input, options = {}) {
    if (typeof input !== 'string') return input;
    
    let sanitized = input;
    
    // Trim whitespace
    if (options.trim !== false) {
      sanitized = sanitized.trim();
    }
    
    // Remove HTML tags
    if (options.stripHtml) {
      sanitized = validator.stripLow(sanitized);
      sanitized = sanitized.replace(/<[^>]*>/g, '');
    }
    
    // Prevent XSS
    if (options.preventXss !== false) {
      sanitized = xss(sanitized, {
        whiteList: options.allowedHtml || {},
        stripIgnoreTag: true,
        stripIgnoreTagBody: ['script']
      });
    }
    
    // Escape special characters
    if (options.escape) {
      sanitized = validator.escape(sanitized);
    }
    
    return sanitized;
  }

  static sanitizeEmail(email) {
    if (typeof email !== 'string') return email;
    
    let sanitized = email.toLowerCase().trim();
    
    if (validator.isEmail(sanitized)) {
      return validator.normalizeEmail(sanitized, {
        gmail_remove_dots: false,
        outlookdotcom_remove_subaddress: false
      });
    }
    
    return sanitized;
  }

  static sanitizePhone(phone) {
    if (typeof phone !== 'string') return phone;
    
    // Remove all non-digit characters except +
    return phone.replace(/[^\d+]/g, '');
  }

  static sanitizeUrl(url) {
    if (typeof url !== 'string') return url;
    
    try {
      const parsed = new URL(url);
      
      // Only allow http and https protocols
      if (!['http:', 'https:'].includes(parsed.protocol)) {
        throw new Error('Invalid protocol');
      }
      
      return parsed.toString();
    } catch {
      return null;
    }
  }

  static sanitizeObject(obj, rules) {
    const sanitized = {};
    
    for (const [key, value] of Object.entries(obj)) {
      const rule = rules[key];
      
      if (rule) {
        switch (rule.type) {
          case 'string':
            sanitized[key] = this.sanitizeString(value, rule.options);
            break;
          case 'email':
            sanitized[key] = this.sanitizeEmail(value);
            break;
          case 'phone':
            sanitized[key] = this.sanitizePhone(value);
            break;
          case 'url':
            sanitized[key] = this.sanitizeUrl(value);
            break;
          default:
            sanitized[key] = value;
        }
      } else if (rule !== false) {
        // Include field if not explicitly excluded
        sanitized[key] = value;
      }
    }
    
    return sanitized;
  }
}

Rate Limiting & Throttling

Validation Rate Limiting

class ValidationRateLimiter {
  constructor() {
    this.attempts = new Map();
    this.blocklist = new Set();
  }

  isBlocked(identifier) {
    return this.blocklist.has(identifier);
  }

  recordAttempt(identifier, success = true) {
    if (!this.attempts.has(identifier)) {
      this.attempts.set(identifier, {
        count: 0,
        firstAttempt: Date.now(),
        lastAttempt: Date.now()
      });
    }

    const record = this.attempts.get(identifier);
    record.count++;
    record.lastAttempt = Date.now();

    if (!success) {
      record.failures = (record.failures || 0) + 1;
      
      // Block after 10 failed validation attempts in 5 minutes
      if (record.failures >= 10 && 
          record.lastAttempt - record.firstAttempt < 300000) {
        this.blocklist.add(identifier);
        
        // Unblock after 30 minutes
        setTimeout(() => {
          this.blocklist.delete(identifier);
          this.attempts.delete(identifier);
        }, 1800000);
      }
    }
  }

  middleware() {
    return (req, res, next) => {
      const identifier = req.ip;
      
      if (this.isBlocked(identifier)) {
        return res.status(429).json({
          error: 'Too many validation failures',
          message: 'Your IP has been temporarily blocked due to repeated validation errors',
          retryAfter: 1800 // seconds
        });
      }
      
      // Store the identifier for later use
      req.validationIdentifier = identifier;
      next();
    };
  }
}

Async Validation

Database Validation

class AsyncValidator {
  static async validateUniqueEmail(email) {
    const existingUser = await User.findOne({ 
      email: email.toLowerCase() 
    });
    
    if (existingUser) {
      throw new Error('Email address is already registered');
    }
    
    return true;
  }

  static async validateUserExists(userId) {
    const user = await User.findById(userId);
    
    if (!user) {
      throw new Error('User not found');
    }
    
    return user;
  }

  static async validatePermissions(userId, resource, action) {
    const user = await User.findById(userId).populate('roles');
    
    const hasPermission = user.roles.some(role => 
      role.permissions.includes(`${resource}:${action}`)
    );
    
    if (!hasPermission) {
      throw new Error('Insufficient permissions');
    }
    
    return true;
  }

  static async validateBusinessRules(data) {
    const validations = [];
    
    // Example: Validate account balance for transactions
    if (data.type === 'withdrawal' && data.amount) {
      validations.push(
        this.validateSufficientBalance(data.accountId, data.amount)
      );
    }
    
    // Example: Validate appointment availability
    if (data.appointmentTime) {
      validations.push(
        this.validateAppointmentSlot(data.providerId, data.appointmentTime)
      );
    }
    
    const results = await Promise.all(validations);
    
    return results.every(result => result === true);
  }

  static async validateSufficientBalance(accountId, amount) {
    const account = await Account.findById(accountId);
    
    if (!account || account.balance < amount) {
      throw new Error('Insufficient account balance');
    }
    
    return true;
  }
}

Best Practices

  • Fail Fast - Validate input as early as possible in the request lifecycle
  • Consistent Errors - Use standardized error formats across your API
  • Sanitize Everything - Never trust user input, always sanitize
  • Rate Limiting - Protect against validation abuse and brute force attacks
  • Async Validation - Use async validators for database checks and external services
  • Performance - Cache validation results where appropriate
  • Documentation - Clearly document validation rules and error responses
🎯 Validation Strategy

Effective request validation combines schema-based validation with custom business rules, proper sanitization, and rate limiting. Always validate on both client and server side, but never rely solely on client-side validation for security.
