Caching Strategies

Implement effective caching strategies to improve performance, reduce server load, and enhance user experience. This guide covers various caching approaches, from browser-level caching to distributed cache systems.

🚀 Performance Impact

Proper caching can reduce page load times by 50-90% and significantly decrease server resource usage. Strategic cache implementation is one of the most effective performance optimizations available.

Browser Caching

Configure HTTP headers to control how browsers cache static assets and dynamic content:

Cache-Control Headers

// Static assets (CSS, JS, images)
Cache-Control: public, max-age=31536000, immutable

// HTML pages
Cache-Control: public, max-age=3600, must-revalidate

// API responses
Cache-Control: private, max-age=300, stale-while-revalidate=60

// Sensitive data
Cache-Control: no-store, no-cache, must-revalidate

ETag Validation

// Generate ETags for content validation
const crypto = require('crypto');

function generateETag(content) {
  return crypto
    .createHash('md5')
    .update(content)
    .digest('hex');
}

// Express middleware example
app.use((req, res, next) => {
  const etag = generateETag(res.content);
  res.set('ETag', etag);
  
  if (req.headers['if-none-match'] === etag) {
    return res.status(304).end();
  }
  
  next();
});

Application Caching

Implement in-memory and persistent caching within your application to reduce database queries and API calls:

Memory Caching

class MemoryCache {
  constructor(maxSize = 1000, ttl = 3600000) {
    this.cache = new Map();
    this.maxSize = maxSize;
    this.defaultTTL = ttl;
  }

  set(key, value, ttl = this.defaultTTL) {
    if (this.cache.size >= this.maxSize) {
      // Remove oldest entry
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }

    const expiry = Date.now() + ttl;
    this.cache.set(key, { value, expiry });
  }

  get(key) {
    const item = this.cache.get(key);
    
    if (!item) return null;
    
    if (Date.now() > item.expiry) {
      this.cache.delete(key);
      return null;
    }
    
    return item.value;
  }

  clear() {
    this.cache.clear();
  }
}

Redis Caching

const redis = require('redis');
const client = redis.createClient();

class RedisCache {
  async set(key, value, ttl = 3600) {
    const serialized = JSON.stringify(value);
    await client.setex(key, ttl, serialized);
  }

  async get(key) {
    const cached = await client.get(key);
    return cached ? JSON.parse(cached) : null;
  }

  async invalidate(pattern) {
    const keys = await client.keys(pattern);
    if (keys.length > 0) {
      await client.del(...keys);
    }
  }

  async getOrSet(key, fetchFn, ttl = 3600) {
    let cached = await this.get(key);
    
    if (cached === null) {
      cached = await fetchFn();
      await this.set(key, cached, ttl);
    }
    
    return cached;
  }
}

Cache Invalidation

// Automated cache invalidation
class CDNManager {
  constructor(apiKey, zone) {
    this.apiKey = apiKey;
    this.zone = zone;
  }

  async purgeCache(urls) {
    const response = await fetch(`/zones/${this.zone}/purge_cache`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ files: urls })
    });

    return response.json();
  }

  async purgeAll() {
    return this.purgeCache(['*']);
  }
}

Cache Patterns

Common caching patterns to handle different scenarios effectively:

Cache-Aside Pattern

async function getCachedData(id) {
  // 1. Check cache first
  let data = await cache.get(`user:${id}`);
  
  if (data) {
    return data; // Cache hit
  }
  
  // 2. Cache miss - fetch from database
  data = await database.getUser(id);
  
  // 3. Update cache
  await cache.set(`user:${id}`, data, 3600);
  
  return data;
}

Write-Through Pattern

async function updateUser(id, updates) {
  // 1. Update database
  const user = await database.updateUser(id, updates);
  
  // 2. Update cache immediately
  await cache.set(`user:${id}`, user, 3600);
  
  return user;
}

Refresh-Ahead Pattern

class RefreshAheadCache {
  constructor(cache, refreshThreshold = 0.8) {
    this.cache = cache;
    this.refreshThreshold = refreshThreshold;
  }

  async get(key, fetchFn, ttl) {
    const item = await this.cache.getWithMetadata(key);
    
    if (!item) {
      // Cache miss
      const value = await fetchFn();
      await this.cache.set(key, value, ttl);
      return value;
    }

    // Check if refresh is needed
    const age = Date.now() - item.timestamp;
    const refreshTime = ttl * this.refreshThreshold;

    if (age > refreshTime) {
      // Async refresh - don't wait
      fetchFn().then(value => {
        this.cache.set(key, value, ttl);
      });
    }

    return item.value;
  }
}

Monitoring & Analytics

Track cache performance metrics to optimize your caching strategy:

Cache Metrics

class CacheMetrics {
  constructor() {
    this.hits = 0;
    this.misses = 0;
    this.sets = 0;
    this.evictions = 0;
  }

  recordHit() {
    this.hits++;
  }

  recordMiss() {
    this.misses++;
  }

  getHitRatio() {
    const total = this.hits + this.misses;
    return total > 0 ? this.hits / total : 0;
  }

  getStats() {
    return {
      hits: this.hits,
      misses: this.misses,
      hitRatio: this.getHitRatio(),
      sets: this.sets,
      evictions: this.evictions
    };
  }
}

Performance Monitoring

// Monitor cache performance
setInterval(async () => {
  const stats = cache.getStats();
  const memoryUsage = process.memoryUsage();
  
  console.log({
    cacheHitRatio: stats.hitRatio,
    cacheSize: stats.size,
    heapUsed: memoryUsage.heapUsed,
    timestamp: new Date()
  });

  // Alert if hit ratio drops below threshold
  if (stats.hitRatio < 0.7) {
    console.warn('Cache hit ratio below optimal threshold');
  }
}, 60000);

Best Practices

  • Cache Key Design - Use consistent, hierarchical naming conventions
  • TTL Strategy - Set appropriate expiration times based on data volatility
  • Memory Management - Monitor cache memory usage and implement size limits
  • Cache Warming - Pre-populate critical cache entries during deployment
  • Graceful Degradation - Handle cache failures without breaking functionality
  • Security Considerations - Never cache sensitive data like passwords or tokens
🎯 Cache Strategy

The most effective caching strategy combines multiple layers: browser caching for static assets, application caching for dynamic data, and CDN caching for global distribution. Monitor performance metrics to fine-tune your approach.
