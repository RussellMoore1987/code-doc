Prometheus Integration

const prometheus = require('prom-client');
const express = require('express');

class MetricsCollector {
  constructor() {
    // Register default metrics (CPU, memory, etc.)
    prometheus.register.setDefaultLabels({
      app: process.env.APP_NAME || 'node-app',
      version: process.env.APP_VERSION || '1.0.0'
    });

    prometheus.collectDefaultMetrics({
      timeout: 10000,
      gcDurationBuckets: [0.001, 0.01, 0.1, 1, 2, 5]
    });

    this.setupCustomMetrics();
  }

  setupCustomMetrics() {
    // HTTP request metrics
    this.httpRequestDuration = new prometheus.Histogram({
      name: 'http_request_duration_seconds',
      help: 'HTTP request duration in seconds',
      labelNames: ['method', 'route', 'status_code'],
      buckets: [0.1, 0.3, 0.5, 0.7, 1, 3, 5, 7, 10]
    });

    this.httpRequestTotal = new prometheus.Counter({
      name: 'http_requests_total',
      help: 'Total HTTP requests',
      labelNames: ['method', 'route', 'status_code']
    });

    // Database metrics
    this.dbQueryDuration = new prometheus.Histogram({
      name: 'database_query_duration_seconds',
      help: 'Database query duration in seconds',
      labelNames: ['operation', 'collection'],
      buckets: [0.01, 0.05, 0.1, 0.2, 0.5, 1, 2, 5]
    });

    this.dbConnections = new prometheus.Gauge({
      name: 'database_connections_active',
      help: 'Number of active database connections'
    });

    // Application metrics
    this.activeUsers = new prometheus.Gauge({
      name: 'active_users_total',
      help: 'Number of currently active users'
    });

    this.backgroundJobs = new prometheus.Gauge({
      name: 'background_jobs_pending',
      help: 'Number of pending background jobs',
      labelNames: ['queue']
    });

    this.cacheHitRate = new prometheus.Gauge({
      name: 'cache_hit_rate',
      help: 'Cache hit rate percentage',
      labelNames: ['cache_type']
    });

    // Business metrics
    this.userRegistrations = new prometheus.Counter({
      name: 'user_registrations_total',
      help: 'Total user registrations'
    });

    this.ordersTotal = new prometheus.Counter({
      name: 'orders_total',
      help: 'Total orders processed',
      labelNames: ['status']
    });

    this.revenueTotal = new prometheus.Counter({
      name: 'revenue_total',
      help: 'Total revenue generated'
    });
  }

  // Express middleware for HTTP metrics
  httpMetricsMiddleware() {
    return (req, res, next) => {
      const start = Date.now();
      const route = this.getRoutePattern(req.route, req.baseUrl);
      
      res.on('finish', () => {
        const duration = (Date.now() - start) / 1000;
        const labels = {
          method: req.method,
          route,
          status_code: res.statusCode
        };

        this.httpRequestDuration.observe(labels, duration);
        this.httpRequestTotal.inc(labels);
      });

      next();
    };
  }

  getRoutePattern(route, baseUrl = '') {
    if (route) {
      return baseUrl + route.path;
    }
    return 'unknown';
  }

  // Database query tracking
  trackDatabaseQuery(operation, collection, duration) {
    this.dbQueryDuration.observe(
      { operation, collection },
      duration / 1000
    );
  }

  updateDatabaseConnections(count) {
    this.dbConnections.set(count);
  }

  // Application metrics
  updateActiveUsers(count) {
    this.activeUsers.set(count);
  }

  updateBackgroundJobs(queue, count) {
    this.backgroundJobs.set({ queue }, count);
  }

  updateCacheHitRate(cacheType, hitRate) {
    this.cacheHitRate.set({ cache_type: cacheType }, hitRate);
  }

  // Business metrics
  recordUserRegistration() {
    this.userRegistrations.inc();
  }

  recordOrder(status) {
    this.ordersTotal.inc({ status });
  }

  recordRevenue(amount) {
    this.revenueTotal.inc(amount);
  }

  // Metrics endpoint
  getMetricsHandler() {
    return async (req, res) => {
      try {
        res.set('Content-Type', prometheus.register.contentType);
        const metrics = await prometheus.register.metrics();
        res.end(metrics);
      } catch (error) {
        res.status(500).end(error.message);
      }
    };
  }
}

class HealthCheckManager {
  constructor() {
    this.checks = new Map();
    this.results = new Map();
    this.checkInterval = 30000; // 30 seconds
    this.timeout = 5000; // 5 seconds
    
    this.setupDefaultChecks();
    this.startHealthChecks();
  }

  setupDefaultChecks() {
    // Database health check
    this.registerCheck('database', async () => {
      const start = Date.now();
      try {
        // Replace with your database ping
        await this.pingDatabase();
        return {
          status: 'healthy',
          responseTime: Date.now() - start,
          details: 'Database connection successful'
        };
      } catch (error) {
        return {
          status: 'unhealthy',
          responseTime: Date.now() - start,
          error: error.message
        };
      }
    });

    // Redis health check
    this.registerCheck('redis', async () => {
      const start = Date.now();
      try {
        // Replace with your Redis ping
        await this.pingRedis();
        return {
          status: 'healthy',
          responseTime: Date.now() - start,
          details: 'Redis connection successful'
        };
      } catch (error) {
        return {
          status: 'unhealthy',
          responseTime: Date.now() - start,
          error: error.message
        };
      }
    });

    // External API health check
    this.registerCheck('external-api', async () => {
      const start = Date.now();
      try {
        const response = await fetch('https://api.external-service.com/health', {
          timeout: 3000
        });
        
        if (response.ok) {
          return {
            status: 'healthy',
            responseTime: Date.now() - start,
            details: `External API responded with ${response.status}`
          };
        } else {
          return {
            status: 'degraded',
            responseTime: Date.now() - start,
            details: `External API responded with ${response.status}`
          };
        }
      } catch (error) {
        return {
          status: 'unhealthy',
          responseTime: Date.now() - start,
          error: error.message
        };
      }
    });

    // File system health check
    this.registerCheck('filesystem', async () => {
      const start = Date.now();
      try {
        const testFile = '/tmp/healthcheck-test';
        const testData = `healthcheck-${Date.now()}`;
        
        await fs.writeFile(testFile, testData);
        const readData = await fs.readFile(testFile, 'utf8');
        await fs.unlink(testFile);
        
        if (readData === testData) {
          return {
            status: 'healthy',
            responseTime: Date.now() - start,
            details: 'File system read/write successful'
          };
        } else {
          return {
            status: 'unhealthy',
            responseTime: Date.now() - start,
            error: 'File system data integrity check failed'
          };
        }
      } catch (error) {
        return {
          status: 'unhealthy',
          responseTime: Date.now() - start,
          error: error.message
        };
      }
    });

    // Memory usage check
    this.registerCheck('memory', async () => {
      const memUsage = process.memoryUsage();
      const totalMem = require('os').totalmem();
      const freeMem = require('os').freemem();
      const usedPercentage = ((totalMem - freeMem) / totalMem) * 100;
      
      let status = 'healthy';
      if (usedPercentage > 90) {
        status = 'unhealthy';
      } else if (usedPercentage > 80) {
        status = 'degraded';
      }
      
      return {
        status,
        responseTime: 0,
        details: {
          heapUsed: `${Math.round(memUsage.heapUsed / 1024 / 1024)}MB`,
          heapTotal: `${Math.round(memUsage.heapTotal / 1024 / 1024)}MB`,
          systemUsage: `${usedPercentage.toFixed(1)}%`
        }
      };
    });
  }

  registerCheck(name, checkFunction, options = {}) {
    this.checks.set(name, {
      check: checkFunction,
      critical: options.critical !== false,
      timeout: options.timeout || this.timeout
    });
  }

  async runCheck(name) {
    const checkConfig = this.checks.get(name);
    if (!checkConfig) {
      throw new Error(`Health check '${name}' not found`);
    }

    try {
      const result = await Promise.race([
        checkConfig.check(),
        new Promise((_, reject) => 
          setTimeout(() => reject(new Error('Health check timeout')), checkConfig.timeout)
        )
      ]);

      return {
        ...result,
        timestamp: new Date().toISOString(),
        critical: checkConfig.critical
      };
    } catch (error) {
      return {
        status: 'unhealthy',
        error: error.message,
        timestamp: new Date().toISOString(),
        critical: checkConfig.critical
      };
    }
  }

  async runAllChecks() {
    const results = {};
    const checkPromises = Array.from(this.checks.keys()).map(async (name) => {
      const result = await this.runCheck(name);
      results[name] = result;
    });

    await Promise.all(checkPromises);
    
    // Determine overall health
    const overallHealth = this.calculateOverallHealth(results);
    
    const healthReport = {
      status: overallHealth,
      timestamp: new Date().toISOString(),
      checks: results,
      uptime: process.uptime(),
      version: process.env.APP_VERSION || '1.0.0'
    };

    this.results.set('latest', healthReport);
    return healthReport;
  }

  calculateOverallHealth(results) {
    const statuses = Object.values(results);
    
    // If any critical check is unhealthy, overall is unhealthy
    const criticalUnhealthy = statuses.some(result => 
      result.critical && result.status === 'unhealthy'
    );
    
    if (criticalUnhealthy) {
      return 'unhealthy';
    }

    // If any check is unhealthy (non-critical), overall is degraded
    const anyUnhealthy = statuses.some(result => result.status === 'unhealthy');
    if (anyUnhealthy) {
      return 'degraded';
    }

    // If any check is degraded, overall is degraded
    const anyDegraded = statuses.some(result => result.status === 'degraded');
    if (anyDegraded) {
      return 'degraded';
    }

    return 'healthy';
  }

  startHealthChecks() {
    // Run initial check
    this.runAllChecks();
    
    // Schedule periodic checks
    this.healthCheckTimer = setInterval(() => {
      this.runAllChecks();
    }, this.checkInterval);
  }

  stopHealthChecks() {
    if (this.healthCheckTimer) {
      clearInterval(this.healthCheckTimer);
    }
  }

  getHealthEndpoint() {
    return async (req, res) => {
      try {
        const health = await this.runAllChecks();
        
        const statusCode = health.status === 'healthy' ? 200 : 
                          health.status === 'degraded' ? 200 : 503;
        
        res.status(statusCode).json(health);
      } catch (error) {
        res.status(503).json({
          status: 'unhealthy',
          error: error.message,
          timestamp: new Date().toISOString()
        });
      }
    };
  }

  getReadinessEndpoint() {
    return async (req, res) => {
      const health = this.results.get('latest');
      
      if (health && health.status !== 'unhealthy') {
        res.status(200).json({ status: 'ready' });
      } else {
        res.status(503).json({ status: 'not ready' });
      }
    };
  }

  getLivenessEndpoint() {
    return (req, res) => {
      // Simple liveness check - if we can respond, we're alive
      res.status(200).json({ 
        status: 'alive',
        timestamp: new Date().toISOString(),
        uptime: process.uptime()
      });
    };
  }

  // Helper methods (implement these based on your infrastructure)
  async pingDatabase() {
    // Implement your database ping logic
    return Promise.resolve();
  }

  async pingRedis() {
    // Implement your Redis ping logic
    return Promise.resolve();
  }
}

Alerting System

Alert Manager Implementation

class AlertManager {
  constructor(options = {}) {
    this.channels = new Map();
    this.rules = new Map();
    this.alertHistory = [];
    this.silencedAlerts = new Set();
    
    this.setupChannels(options.channels || {});
  }

  setupChannels(channelConfigs) {
    // Email channel
    if (channelConfigs.email) {
      this.channels.set('email', new EmailAlertChannel(channelConfigs.email));
    }

    // Slack channel
    if (channelConfigs.slack) {
      this.channels.set('slack', new SlackAlertChannel(channelConfigs.slack));
    }

    // PagerDuty channel
    if (channelConfigs.pagerduty) {
      this.channels.set('pagerduty', new PagerDutyAlertChannel(channelConfigs.pagerduty));
    }

    // Webhook channel
    if (channelConfigs.webhook) {
      this.channels.set('webhook', new WebhookAlertChannel(channelConfigs.webhook));
    }
  }

  addRule(name, condition, options = {}) {
    this.rules.set(name, {
      condition,
      severity: options.severity || 'warning',
      channels: options.channels || ['email'],
      cooldown: options.cooldown || 300000, // 5 minutes
      threshold: options.threshold || 1,
      lastTriggered: null,
      triggerCount: 0,
      description: options.description || name
    });
  }

  async checkAlerts(metrics) {
    for (const [ruleName, rule] of this.rules) {
      try {
        const shouldAlert = await this.evaluateRule(rule, metrics);
        
        if (shouldAlert && !this.isInCooldown(rule) && !this.isSilenced(ruleName)) {
          await this.triggerAlert(ruleName, rule, metrics);
        }
      } catch (error) {
        console.error(`Error evaluating alert rule ${ruleName}:`, error);
      }
    }
  }

  async evaluateRule(rule, metrics) {
    try {
      return await rule.condition(metrics);
    } catch (error) {
      console.error('Error in alert rule condition:', error);
      return false;
    }
  }

  isInCooldown(rule) {
    if (!rule.lastTriggered) return false;
    return (Date.now() - rule.lastTriggered) < rule.cooldown;
  }

  isSilenced(ruleName) {
    return this.silencedAlerts.has(ruleName);
  }

  async triggerAlert(ruleName, rule, metrics) {
    const alert = {
      id: this.generateAlertId(),
      name: ruleName,
      description: rule.description,
      severity: rule.severity,
      timestamp: new Date().toISOString(),
      metrics: this.sanitizeMetrics(metrics),
      status: 'firing'
    };

    // Update rule state
    rule.lastTriggered = Date.now();
    rule.triggerCount++;

    // Store alert
    this.alertHistory.push(alert);

    // Send notifications
    await this.sendNotifications(alert, rule.channels);

    console.log(`Alert triggered: ${ruleName} (${rule.severity})`);
  }

  async sendNotifications(alert, channels) {
    const notifications = channels.map(async (channelName) => {
      const channel = this.channels.get(channelName);
      if (channel) {
        try {
          await channel.send(alert);
        } catch (error) {
          console.error(`Failed to send alert to ${channelName}:`, error);
        }
      }
    });

    await Promise.allSettled(notifications);
  }

  silenceAlert(ruleName, duration = 3600000) { // 1 hour default
    this.silencedAlerts.add(ruleName);
    
    setTimeout(() => {
      this.silencedAlerts.delete(ruleName);
    }, duration);
  }

  unsilenceAlert(ruleName) {
    this.silencedAlerts.delete(ruleName);
  }

  generateAlertId() {
    return Date.now().toString(36) + Math.random().toString(36).substr(2);
  }

  sanitizeMetrics(metrics) {
    // Remove sensitive data from metrics before storing/sending
    const sanitized = { ...metrics };
    delete sanitized.credentials;
    delete sanitized.tokens;
    return sanitized;
  }

  getAlertHistory(limit = 100) {
    return this.alertHistory
      .slice(-limit)
      .sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp));
  }

  getActiveAlerts() {
    return this.alertHistory.filter(alert => alert.status === 'firing');
  }
}

// Alert channel implementations
class EmailAlertChannel {
  constructor(config) {
    this.config = config;
    this.nodemailer = require('nodemailer');
    this.transporter = this.nodemailer.createTransporter(config.smtp);
  }

  async send(alert) {
    const subject = `${alert.severity.toUpperCase()}: ${alert.name}`;
    const html = this.formatAlertEmail(alert);

    await this.transporter.sendMail({
      from: this.config.from,
      to: this.config.to,
      subject,
      html
    });
  }

  formatAlertEmail(alert) {
    return `
      
Alert: ${alert.name}

      
Severity: ${alert.severity}

      
Time: ${alert.timestamp}

      
Description: ${alert.description}

      
${JSON.stringify(alert.metrics, null, 2)}

    `;
  }
}

class SlackAlertChannel {
  constructor(config) {
    this.webhookUrl = config.webhookUrl;
    this.channel = config.channel;
  }

  async send(alert) {
    const color = this.getSeverityColor(alert.severity);
    const payload = {
      channel: this.channel,
      attachments: [{
        color,
        title: `Alert: ${alert.name}`,
        fields: [
          { title: 'Severity', value: alert.severity, short: true },
          { title: 'Time', value: alert.timestamp, short: true },
          { title: 'Description', value: alert.description, short: false }
        ],
        ts: Math.floor(new Date(alert.timestamp).getTime() / 1000)
      }]
    };

    await fetch(this.webhookUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });
  }

  getSeverityColor(severity) {
    const colors = {
      critical: '#ff0000',
      warning: '#ffaa00',
      info: '#36a64f'
    };
    return colors[severity] || '#808080';
  }
}

Distributed Tracing

Jaeger Tracing Implementation

const { initTracer } = require('jaeger-client');
const opentracing = require('opentracing');

class TracingManager {
  constructor(serviceName, config = {}) {
    this.serviceName = serviceName;
    this.tracer = this.initializeTracer(config);
    opentracing.initGlobalTracer(this.tracer);
  }

  initializeTracer(config) {
    const defaultConfig = {
      serviceName: this.serviceName,
      sampler: {
        type: 'const',
        param: 1 // Sample all traces
      },
      reporter: {
        logSpans: true,
        agentHost: process.env.JAEGER_AGENT_HOST || 'localhost',
        agentPort: parseInt(process.env.JAEGER_AGENT_PORT) || 6832
      }
    };

    const finalConfig = { ...defaultConfig, ...config };
    return initTracer(finalConfig);
  }

  // Express middleware for automatic request tracing
  expressMiddleware() {
    return (req, res, next) => {
      const span = this.tracer.startSpan(`HTTP ${req.method} ${req.path}`);
      
      // Add request tags
      span.setTag('http.method', req.method);
      span.setTag('http.url', req.originalUrl);
      span.setTag('user.id', req.user?.id);
      
      // Store span in request context
      req.span = span;
      
      // Trace response
      res.on('finish', () => {
        span.setTag('http.status_code', res.statusCode);
        
        if (res.statusCode >= 400) {
          span.setTag('error', true);
          span.setTag('error.message', `HTTP ${res.statusCode}`);
        }
        
        span.finish();
      });
      
      next();
    };
  }

  // Database query tracing wrapper
  traceDatabase(operation, collection) {
    return async (queryFunction) => {
      const span = this.tracer.startSpan(`DB ${operation}`, {
        childOf: this.getCurrentSpan()
      });
      
      span.setTag('db.type', 'mongodb');
      span.setTag('db.operation', operation);
      span.setTag('db.collection', collection);
      
      try {
        const result = await queryFunction();
        span.setTag('db.success', true);
        return result;
      } catch (error) {
        span.setTag('error', true);
        span.setTag('error.message', error.message);
        throw error;
      } finally {
        span.finish();
      }
    };
  }

  // External API call tracing
  traceExternalCall(serviceName, operation) {
    return async (callFunction) => {
      const span = this.tracer.startSpan(`External ${serviceName}`, {
        childOf: this.getCurrentSpan()
      });
      
      span.setTag('component', 'http-client');
      span.setTag('peer.service', serviceName);
      span.setTag('operation', operation);
      
      try {
        const result = await callFunction();
        span.setTag('success', true);
        return result;
      } catch (error) {
        span.setTag('error', true);
        span.setTag('error.message', error.message);
        throw error;
      } finally {
        span.finish();
      }
    };
  }

  // Background job tracing
  traceBackgroundJob(jobType, jobId) {
    const span = this.tracer.startSpan(`Job ${jobType}`);
    
    span.setTag('job.type', jobType);
    span.setTag('job.id', jobId);
    span.setTag('component', 'background-worker');
    
    return {
      span,
      finish: (success = true, error = null) => {
        span.setTag('job.success', success);
        if (error) {
          span.setTag('error', true);
          span.setTag('error.message', error.message);
        }
        span.finish();
      }
    };
  }

  // Custom span creation
  createSpan(operationName, options = {}) {
    return this.tracer.startSpan(operationName, {
      childOf: options.parentSpan || this.getCurrentSpan(),
      tags: options.tags || {}
    });
  }

  getCurrentSpan() {
    return this.tracer.getCurrentSpan?.() || null;
  }

  // Distributed context propagation
  injectContext(span, format, carrier) {
    this.tracer.inject(span.context(), format, carrier);
  }

  extractContext(format, carrier) {
    return this.tracer.extract(format, carrier);
  }

  // Correlation ID management
  generateCorrelationId() {
    return Math.random().toString(36).substring(2) + Date.now().toString(36);
  }

  setCorrelationId(span, correlationId) {
    span.setTag('correlation.id', correlationId);
  }

  // Error reporting with tracing context
  reportError(error, span = null) {
    const currentSpan = span || this.getCurrentSpan();
    
    if (currentSpan) {
      currentSpan.setTag('error', true);
      currentSpan.setTag('error.message', error.message);
      currentSpan.setTag('error.stack', error.stack);
      
      currentSpan.log({
        event: 'error',
        message: error.message,
        stack: error.stack,
        timestamp: Date.now()
      });
    }
    
    // Also send to error tracking service
    console.error('Traced Error:', error);
  }

  close() {
    if (this.tracer.close) {
      this.tracer.close();
    }
  }
}

// Usage example
const tracingManager = new TracingManager('user-service');

// In Express app

class TracedUserRepository {

  }

      return await User.findById(userId);

      return await User.create(userData);
    });
  }
}

Best Practices

  • Four Golden Signals - Monitor latency, traffic, errors, and saturation
  • SLIs and SLOs - Define Service Level Indicators and Objectives
  • Alert Fatigue - Minimize false positives and noise
  • Observability - Combine metrics, logs, and traces for complete visibility
  • Business Metrics - Track metrics that matter to the business
  • Automation - Automate monitoring setup and alerting
  • Documentation - Document monitoring procedures and runbooks
🎯 Monitoring Excellence

Effective monitoring systems balance comprehensive coverage with actionable insights. Focus on metrics that indicate real problems, implement proper alerting thresholds, and ensure your monitoring system itself is reliable and well-monitored.
