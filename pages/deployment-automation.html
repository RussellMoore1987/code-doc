CI/CD Pipeline Setup

GitHub Actions Deployment Workflow

# .github/workflows/deploy.yml
name: Deploy Application

on:
  push:
    branches: [ main, staging ]
  pull_request:
    branches: [ main ]

env:
  NODE_VERSION: '18'
  DOCKER_REGISTRY: ghcr.io
  IMAGE_NAME: ${{ github.repository }}

jobs:
  test:
    runs-on: ubuntu-latest
    
    services:
      postgres:
        image: postgres:14
        env:
          POSTGRES_PASSWORD: postgres
          POSTGRES_DB: test
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
        ports:
          - 5432:5432

    steps:
    - name: Checkout code
      uses: actions/checkout@v4

    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: ${{ env.NODE_VERSION }}
        cache: 'npm'

    - name: Install dependencies
      run: |
        npm ci
        npm audit --audit-level high

    - name: Run linting
      run: npm run lint

    - name: Run unit tests
      run: npm run test:unit
      env:
        NODE_ENV: test
        DATABASE_URL: postgresql://postgres:postgres@localhost:5432/test

    - name: Run integration tests
      run: npm run test:integration
      env:
        NODE_ENV: test
        DATABASE_URL: postgresql://postgres:postgres@localhost:5432/test

    - name: Build application
      run: npm run build

    - name: Upload build artifacts
      uses: actions/upload-artifact@v4
      with:
        name: build-files
        path: dist/

  security-scan:
    runs-on: ubuntu-latest
    needs: test
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4

    - name: Run Trivy vulnerability scanner
      uses: aquasecurity/trivy-action@master
      with:
        scan-type: 'fs'
        scan-ref: '.'
        format: 'sarif'
        output: 'trivy-results.sarif'

    - name: Upload Trivy scan results
      uses: github/codeql-action/upload-sarif@v3
      with:
        sarif_file: 'trivy-results.sarif'

    - name: Run CodeQL Analysis
      uses: github/codeql-action/analyze@v3

  build-and-push:
    runs-on: ubuntu-latest
    needs: [test, security-scan]
    if: github.event_name == 'push'
    
    permissions:
      contents: read
      packages: write

    steps:
    - name: Checkout code
      uses: actions/checkout@v4

    - name: Setup Docker Buildx
      uses: docker/setup-buildx-action@v3

    - name: Login to Container Registry
      uses: docker/login-action@v3
      with:
        registry: ${{ env.DOCKER_REGISTRY }}
        username: ${{ github.actor }}
        password: ${{ secrets.GITHUB_TOKEN }}

    - name: Extract metadata
      id: meta
      uses: docker/metadata-action@v5
      with:
        images: ${{ env.DOCKER_REGISTRY }}/${{ env.IMAGE_NAME }}
        tags: |
          type=ref,event=branch
          type=ref,event=pr
          type=sha,prefix={{branch}}-
          type=raw,value=latest,enable={{is_default_branch}}

    - name: Build and push Docker image
      uses: docker/build-push-action@v5
      with:
        context: .
        push: true
        tags: ${{ steps.meta.outputs.tags }}
        labels: ${{ steps.meta.outputs.labels }}
        cache-from: type=gha
        cache-to: type=gha,mode=max

  deploy-staging:
    runs-on: ubuntu-latest
    needs: build-and-push
    if: github.ref == 'refs/heads/staging'
    environment: staging

    steps:
    - name: Deploy to staging
      run: |
        echo "Deploying to staging environment"
        # Add your staging deployment commands here

  deploy-production:
    runs-on: ubuntu-latest
    needs: build-and-push
    if: github.ref == 'refs/heads/main'
    environment: production

    steps:
    - name: Deploy to production
      run: |
        echo "Deploying to production environment"
        # Add your production deployment commands here

    - name: Run smoke tests
      run: |
        echo "Running post-deployment smoke tests"
        # Add smoke test commands

Multi-stage Dockerfile

# Dockerfile
# Build stage
FROM node:18-alpine AS builder

WORKDIR /app

# Copy package files
COPY package*.json ./

# Install dependencies
RUN npm ci --only=production && npm cache clean --force

# Copy source code
COPY . .

# Build application
RUN npm run build

# Production stage
FROM node:18-alpine AS production

# Install security updates
RUN apk update && apk upgrade && apk add --no-cache dumb-init

# Create non-root user
RUN addgroup -g 1001 -S nodejs
RUN adduser -S nextjs -u 1001

# Set working directory
WORKDIR /app

# Copy built application from builder stage
COPY --from=builder --chown=nextjs:nodejs /app/dist ./dist
COPY --from=builder --chown=nextjs:nodejs /app/node_modules ./node_modules
COPY --from=builder --chown=nextjs:nodejs /app/package.json ./package.json

# Switch to non-root user
USER nextjs

# Expose port
EXPOSE 3000

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD curl -f http://localhost:3000/health || exit 1

# Use dumb-init to handle signals properly
ENTRYPOINT ["dumb-init", "--"]

# Start application
CMD ["node", "dist/server.js"]

Deployment Scripts

Automated Deployment Script

#!/bin/bash

# deploy.sh - Automated deployment script
set -euo pipefail

# Configuration
PROJECT_NAME="myapp"
DOCKER_REGISTRY="ghcr.io/myorg"
KUBECTL_CONTEXT="myapp-cluster"
NAMESPACE="myapp"

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# Logging functions
log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# Check prerequisites
check_prerequisites() {
    log_info "Checking prerequisites..."
    
    if ! command -v docker &> /dev/null; then
        log_error "Docker is not installed"
        exit 1
    fi
    
    if ! command -v kubectl &> /dev/null; then
        log_error "kubectl is not installed"
        exit 1
    fi
    
    if ! command -v helm &> /dev/null; then
        log_warn "Helm is not installed, some features may not work"
    fi
    
    log_info "Prerequisites check passed"
}

# Build and push Docker image
build_and_push() {
    local tag=${1:-latest}
    local image_name="${DOCKER_REGISTRY}/${PROJECT_NAME}:${tag}"
    
    log_info "Building Docker image: ${image_name}"
    
    # Build image
    docker build -t "${image_name}" .
    
    # Run security scan
    if command -v trivy &> /dev/null; then
        log_info "Running security scan..."
        trivy image "${image_name}"
    fi
    
    # Push image
    log_info "Pushing Docker image..."
    docker push "${image_name}"
    
    echo "${image_name}"
}

# Deploy to Kubernetes
deploy_to_k8s() {
    local image_name=$1
    local environment=${2:-production}
    
    log_info "Deploying to Kubernetes (${environment})..."
    
    # Set kubectl context
    kubectl config use-context "${KUBECTL_CONTEXT}"
    
    # Create namespace if it doesn't exist
    kubectl create namespace "${NAMESPACE}" --dry-run=client -o yaml | kubectl apply -f -
    
    # Update image in deployment
    kubectl set image deployment/app-deployment \
        app="${image_name}" \
        -n "${NAMESPACE}"
    
    # Wait for rollout
    log_info "Waiting for rollout to complete..."
    kubectl rollout status deployment/app-deployment -n "${NAMESPACE}" --timeout=600s
    
    # Verify deployment
    verify_deployment
}

# Verify deployment
verify_deployment() {
    log_info "Verifying deployment..."
    
    # Check pod status
    local ready_pods=$(kubectl get pods -n "${NAMESPACE}" -l app="${PROJECT_NAME}" \
        -o jsonpath='{.items[*].status.containerStatuses[0].ready}' | grep -o "true" | wc -l)
    local total_pods=$(kubectl get pods -n "${NAMESPACE}" -l app="${PROJECT_NAME}" \
        --no-headers | wc -l)
    
    if [ "${ready_pods}" -eq "${total_pods}" ] && [ "${total_pods}" -gt 0 ]; then
        log_info "All ${total_pods} pods are ready"
    else
        log_error "Only ${ready_pods}/${total_pods} pods are ready"
        kubectl get pods -n "${NAMESPACE}" -l app="${PROJECT_NAME}"
        exit 1
    fi
    
    # Run health check
    run_health_check
}

# Run health check
run_health_check() {
    log_info "Running health check..."
    
    local service_url=$(kubectl get ingress app-ingress -n "${NAMESPACE}" \
        -o jsonpath='{.spec.rules[0].host}')
    
    if [ -n "${service_url}" ]; then
        local health_url="https://${service_url}/health"
        
        # Wait for service to be ready
        for i in {1..30}; do
            if curl -sf "${health_url}" > /dev/null; then
                log_info "Health check passed: ${health_url}"
                return 0
            fi
            log_info "Health check attempt ${i}/30 failed, retrying..."
            sleep 10
        done
        
        log_error "Health check failed after 30 attempts"
        exit 1
    else
        log_warn "No ingress found, skipping external health check"
    fi
}

# Rollback deployment
rollback_deployment() {
    log_warn "Rolling back deployment..."
    
    kubectl rollout undo deployment/app-deployment -n "${NAMESPACE}"
    kubectl rollout status deployment/app-deployment -n "${NAMESPACE}" --timeout=600s
    
    log_info "Rollback completed"
}

# Blue-green deployment
blue_green_deploy() {
    local image_name=$1
    local current_deployment="app-deployment"
    local new_deployment="app-deployment-green"
    
    log_info "Starting blue-green deployment..."
    
    # Create green deployment
    kubectl get deployment "${current_deployment}" -n "${NAMESPACE}" -o yaml | \
        sed "s/${current_deployment}/${new_deployment}/g" | \
        sed "s/app: ${PROJECT_NAME}/app: ${PROJECT_NAME}-green/g" | \
        kubectl apply -f -
    
    # Update green deployment image
    kubectl set image deployment/"${new_deployment}" \
        app="${image_name}" \
        -n "${NAMESPACE}"
    
    # Wait for green deployment
    kubectl rollout status deployment/"${new_deployment}" -n "${NAMESPACE}" --timeout=600s
    
    # Switch traffic to green
    kubectl patch service app-service -n "${NAMESPACE}" \
        -p '{"spec":{"selector":{"app":"'${PROJECT_NAME}'-green"}}}'
    
    log_info "Traffic switched to green deployment"
    
    # Cleanup old deployment after verification
    read -p "Delete blue deployment? (y/N): " -n 1 -r
    echo
    if [[ $REPLY =~ ^[Yy]$ ]]; then
        kubectl delete deployment "${current_deployment}" -n "${NAMESPACE}"
        kubectl patch service app-service -n "${NAMESPACE}" \
            -p '{"spec":{"selector":{"app":"'${PROJECT_NAME}'"}}}'
        log_info "Blue deployment cleaned up"
    fi
}

# Main deployment function
main() {
    local environment=${1:-production}
    local deployment_type=${2:-rolling}
    local tag=${3:-$(git rev-parse --short HEAD)}
    
    log_info "Starting deployment (env: ${environment}, type: ${deployment_type}, tag: ${tag})"
    
    check_prerequisites
    
    # Build and push image
    local image_name=$(build_and_push "${tag}")
    
    # Deploy based on strategy
    case "${deployment_type}" in
        "rolling")
            deploy_to_k8s "${image_name}" "${environment}"
            ;;
        "blue-green")
            blue_green_deploy "${image_name}"
            ;;
        *)
            log_error "Unknown deployment type: ${deployment_type}"
            exit 1
            ;;
    esac
    
    log_info "Deployment completed successfully!"
}

# Handle script arguments
case "${1:-deploy}" in
    "deploy")
        main "${2:-production}" "${3:-rolling}" "${4:-$(git rev-parse --short HEAD)}"
        ;;
    "rollback")
        rollback_deployment
        ;;
    "health")
        run_health_check
        ;;
    *)
        echo "Usage: $0 {deploy|rollback|health} [environment] [deployment-type] [tag]"
        exit 1
        ;;
esac

Deployment Monitoring Setup

class DeploymentMonitor {
  constructor(config) {
    this.config = config;
    this.metrics = new Map();
    this.alerts = new Map();
    this.deploymentHistory = [];
  }

  // Monitor deployment progress
  async monitorDeployment(deploymentId, strategy = 'rolling') {
    const deployment = {
      id: deploymentId,
      strategy,
      startTime: new Date(),
      status: 'in-progress',
      metrics: {
        podsReady: 0,
        podsTotal: 0,
        healthChecksPassed: 0,
        errorRate: 0,
        responseTime: 0
      }
    };

    this.deploymentHistory.push(deployment);

    try {
      // Monitor based on strategy
      switch (strategy) {
        case 'rolling':
          await this.monitorRollingDeployment(deployment);
          break;
        case 'blue-green':
          await this.monitorBlueGreenDeployment(deployment);
          break;
        case 'canary':
          await this.monitorCanaryDeployment(deployment);
          break;
      }

      deployment.status = 'completed';
      deployment.endTime = new Date();
      
    } catch (error) {
      deployment.status = 'failed';
      deployment.error = error.message;
      deployment.endTime = new Date();
      
      await this.handleDeploymentFailure(deployment);
      throw error;
    }

    return deployment;
  }

  // Monitor rolling deployment
  async monitorRollingDeployment(deployment) {
    const maxWaitTime = 10 * 60 * 1000; // 10 minutes
    const checkInterval = 10000; // 10 seconds
    const startTime = Date.now();

    while (Date.now() - startTime < maxWaitTime) {
      const status = await this.checkDeploymentStatus(deployment.id);
      
      deployment.metrics = {
        ...deployment.metrics,
        ...status
      };

      if (status.podsReady === status.podsTotal && status.podsTotal > 0) {
        // All pods are ready, check health
        const healthCheck = await this.performHealthCheck();
        
        if (healthCheck.success) {
          console.log('Rolling deployment completed successfully');
          return;
        }
      }

      // Check for deployment issues
      if (status.errorRate > 0.1) { // 10% error rate threshold
        throw new Error(`High error rate detected: ${status.errorRate}`);
      }

      await this.sleep(checkInterval);
    }

    throw new Error('Deployment timeout');
  }

  // Monitor blue-green deployment
  async monitorBlueGreenDeployment(deployment) {
    console.log('Monitoring blue-green deployment...');
    
    // Phase 1: Deploy green environment
    await this.monitorGreenEnvironmentDeployment(deployment);
    
    // Phase 2: Health check green environment
    await this.validateGreenEnvironment(deployment);
    
    // Phase 3: Switch traffic
    await this.switchTrafficToGreen(deployment);
    
    // Phase 4: Monitor post-switch
    await this.monitorPostSwitch(deployment);
  }

  // Monitor canary deployment
  async monitorCanaryDeployment(deployment) {
    const stages = [
      { traffic: 10, duration: 5 * 60 * 1000 },  // 10% for 5 minutes
      { traffic: 25, duration: 10 * 60 * 1000 }, // 25% for 10 minutes
      { traffic: 50, duration: 15 * 60 * 1000 }, // 50% for 15 minutes
      { traffic: 100, duration: 5 * 60 * 1000 }  // 100% for 5 minutes
    ];

    for (const stage of stages) {
      console.log(`Canary stage: ${stage.traffic}% traffic`);
      
      await this.updateCanaryTraffic(deployment.id, stage.traffic);
      await this.monitorCanaryStage(deployment, stage);
      
      // Check metrics before proceeding
      const metrics = await this.getCanaryMetrics(deployment.id);
      if (!this.validateCanaryMetrics(metrics)) {
        throw new Error('Canary metrics validation failed');
      }
    }
  }

  // Check deployment status
  async checkDeploymentStatus(deploymentId) {
    // This would integrate with your container orchestrator
    // Example for Kubernetes:
    const kubectl = require('kubectl');
    
    const deployment = await kubectl.get('deployment', deploymentId);
    const pods = await kubectl.get('pods', { label: `app=${deploymentId}` });
    
    const podsReady = pods.items.filter(pod => 
      pod.status.containerStatuses?.every(c => c.ready)
    ).length;
    
    return {
      podsReady,
      podsTotal: pods.items.length,
      replicas: deployment.spec.replicas,
      readyReplicas: deployment.status.readyReplicas || 0,
      updatedReplicas: deployment.status.updatedReplicas || 0
    };
  }

  // Perform health check
  async performHealthCheck() {
    try {
      const response = await fetch(`${this.config.healthCheckUrl}/health`);
      const data = await response.json();
      
      return {
        success: response.ok && data.status === 'healthy',
        response: data,
        statusCode: response.status
      };
    } catch (error) {
      return {
        success: false,
        error: error.message
      };
    }
  }

  // Handle deployment failure
  async handleDeploymentFailure(deployment) {
    console.error(`Deployment ${deployment.id} failed:`, deployment.error);
    
    // Send alerts
    await this.sendAlert({
      type: 'deployment_failure',
      deployment: deployment.id,
      error: deployment.error,
      timestamp: new Date()
    });

    // Auto-rollback if enabled
    if (this.config.autoRollback) {
      try {
        await this.rollbackDeployment(deployment.id);
        console.log('Automatic rollback completed');
      } catch (rollbackError) {
        console.error('Rollback failed:', rollbackError);
        await this.sendAlert({
          type: 'rollback_failure',
          deployment: deployment.id,
          error: rollbackError.message,
          timestamp: new Date()
        });
      }
    }
  }

  // Rollback deployment
  async rollbackDeployment(deploymentId) {
    // Implementation depends on your deployment system
    console.log(`Rolling back deployment: ${deploymentId}`);
    
    // Example for Kubernetes
    // kubectl rollout undo deployment/${deploymentId}
  }

  // Send alert
  async sendAlert(alert) {
    // Integration with alerting system (Slack, PagerDuty, etc.)
    console.log('Alert:', alert);
    
    if (this.config.slackWebhook) {
      await this.sendSlackAlert(alert);
    }
    
    if (this.config.emailAlerts) {
      await this.sendEmailAlert(alert);
    }
  }

  // Get deployment metrics
  getDeploymentMetrics(timeRange = '1h') {
    return this.deploymentHistory
      .filter(d => new Date() - d.startTime < this.parseTimeRange(timeRange))
      .map(d => ({
        id: d.id,
        strategy: d.strategy,
        duration: d.endTime ? d.endTime - d.startTime : null,
        status: d.status,
        success: d.status === 'completed'
      }));
  }

  // Utility functions
  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  parseTimeRange(range) {
    const unit = range.slice(-1);
    const value = parseInt(range.slice(0, -1));
    
    switch (unit) {
      case 'h': return value * 60 * 60 * 1000;
      case 'm': return value * 60 * 1000;
      case 's': return value * 1000;
      default: return value;
    }
  }
}
