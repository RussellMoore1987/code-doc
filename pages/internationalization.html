Internationalization (i18n)

Implement comprehensive internationalization support to make your application accessible to users worldwide. This guide covers locale management, text translation, formatting, and cultural adaptation strategies.

🌐 Global Reach

Internationalization enables your application to support multiple languages, regions, and cultural conventions without code changes. Proper i18n implementation can significantly expand your user base and market reach.

I18n Setup

Set up a robust internationalization framework that supports dynamic language switching and fallback mechanisms:

// i18n.js - Configuration

const Backend = require('i18next-fs-backend');
const middleware = require('i18next-http-middleware');

i18next

  .use(middleware.LanguageDetector)

    backend: {
      loadPath: './locales/{{lng}}/{{ns}}.json',
      addPath: './locales/{{lng}}/{{ns}}.missing.json'
    },
    
    detection: {
      order: ['querystring', 'cookie', 'header'],
      caches: ['cookie'],
      lookupQuerystring: 'lang',
      lookupCookie: 'i18next',
      cookieMinutes: 10080, // 7 days
    },
    
    interpolation: {
      escapeValue: false // React already escapes
    },
    
    saveMissing: true,
    saveMissingTo: 'fallback'
  });

module.exports = i18next;

Vanilla JavaScript Implementation

class I18nManager {
  constructor() {
    this.currentLocale = 'en';
    this.translations = {};
    this.fallbackLocale = 'en';
  }

  async loadTranslations(locale) {
    if (this.translations[locale]) {
      return this.translations[locale];
    }

    try {
      const response = await fetch(`/api/translations/${locale}`);
      const data = await response.json();
      this.translations[locale] = data;
      return data;
    } catch (error) {
      console.error(`Failed to load translations for ${locale}:`, error);
      return this.translations[this.fallbackLocale] || {};
    }
  }

  async setLocale(locale) {
    await this.loadTranslations(locale);
    this.currentLocale = locale;
    this.updateDOM();
    this.savePreference(locale);
  }

  t(key, params = {}) {
    const translation = this.getTranslation(key, this.currentLocale) ||
                       this.getTranslation(key, this.fallbackLocale) ||
                       key;

    return this.interpolate(translation, params);
  }

  getTranslation(key, locale) {
    const keys = key.split('.');
    let value = this.translations[locale];

    for (const k of keys) {
      if (value && typeof value === 'object') {
        value = value[k];
      } else {
        return null;
      }
    }

    return value;
  }

  interpolate(template, params) {
    return template.replace(/\{\{(\w+)\}\}/g, (match, key) => {
      return params[key] || match;
    });
  }

  updateDOM() {
    document.querySelectorAll('[data-i18n]').forEach(element => {
      const key = element.getAttribute('data-i18n');
      element.textContent = this.t(key);
    });

    document.querySelectorAll('[data-i18n-placeholder]').forEach(element => {
      const key = element.getAttribute('data-i18n-placeholder');
      element.placeholder = this.t(key);
    });
  }

  savePreference(locale) {
    localStorage.setItem('preferred-language', locale);
    document.cookie = `lang=${locale}; path=/; max-age=604800`; // 7 days
  }
}

Number & Date Formatting

Number Formatting

class NumberFormatter {
  constructor(locale = 'en') {
    this.locale = locale;
  }

  formatCurrency(amount, currency = 'USD') {
    return new Intl.NumberFormat(this.locale, {
      style: 'currency',
      currency: currency,
      minimumFractionDigits: 2
    }).format(amount);
  }

  formatPercent(value) {
    return new Intl.NumberFormat(this.locale, {
      style: 'percent',
      minimumFractionDigits: 1,
      maximumFractionDigits: 2
    }).format(value);
  }

  formatNumber(number, options = {}) {
    return new Intl.NumberFormat(this.locale, {
      minimumFractionDigits: 0,
      maximumFractionDigits: 2,
      ...options
    }).format(number);
  }

  formatFileSize(bytes) {
    const sizes = ['bytes', 'KB', 'MB', 'GB', 'TB'];
    if (bytes === 0) return '0 bytes';
    
    const i = Math.floor(Math.log(bytes) / Math.log(1024));
    const size = bytes / Math.pow(1024, i);
    
    return `${this.formatNumber(size, { maximumFractionDigits: 1 })} ${sizes[i]}`;
  }
}

Date & Time Formatting

class DateTimeFormatter {
  constructor(locale = 'en', timeZone = 'UTC') {
    this.locale = locale;
    this.timeZone = timeZone;
  }

  formatDate(date, style = 'medium') {
    const options = {
      timeZone: this.timeZone,
      dateStyle: style
    };
    
    return new Intl.DateTimeFormat(this.locale, options).format(date);
  }

  formatTime(date, style = 'short') {
    const options = {
      timeZone: this.timeZone,
      timeStyle: style
    };
    
    return new Intl.DateTimeFormat(this.locale, options).format(date);
  }

  formatDateTime(date, options = {}) {
    const defaultOptions = {
      timeZone: this.timeZone,
      year: 'numeric',
      month: 'long',
      day: 'numeric',
      hour: '2-digit',
      minute: '2-digit',
      ...options
    };
    
    return new Intl.DateTimeFormat(this.locale, defaultOptions).format(date);
  }

  formatRelativeTime(date) {
    const rtf = new Intl.RelativeTimeFormat(this.locale, { numeric: 'auto' });
    const diff = date.getTime() - Date.now();
    const absDiff = Math.abs(diff);

    if (absDiff < 1000 * 60) {
      return rtf.format(Math.round(diff / 1000), 'second');
    } else if (absDiff < 1000 * 60 * 60) {
      return rtf.format(Math.round(diff / (1000 * 60)), 'minute');
    } else if (absDiff < 1000 * 60 * 60 * 24) {
      return rtf.format(Math.round(diff / (1000 * 60 * 60)), 'hour');
    } else if (absDiff < 1000 * 60 * 60 * 24 * 30) {
      return rtf.format(Math.round(diff / (1000 * 60 * 60 * 24)), 'day');
    } else {
      return this.formatDate(date);
    }
  }
}

Pluralization

Handle complex pluralization rules for different languages: 

// Pluralization rules
const pluralizationRules = {
  en: {
    cardinal: ['one', 'other'],
    ordinal: ['one', 'two', 'few', 'other']
  },
  ru: {
    cardinal: ['one', 'few', 'many', 'other'],
    ordinal: ['other']
  },
  ar: {
    cardinal: ['zero', 'one', 'two', 'few', 'many', 'other'],
    ordinal: ['other']
  }
};

class PluralFormatter {
  constructor(locale) {
    this.locale = locale;
    this.pr = new Intl.PluralRules(locale, { type: 'cardinal' });
    this.prOrdinal = new Intl.PluralRules(locale, { type: 'ordinal' });
  }

  formatPlural(count, translations) {
    const rule = this.pr.select(count);
    return translations[rule] || translations.other || '';
  }

  formatOrdinal(number) {
    const rule = this.prOrdinal.select(number);
    const suffixes = {
      one: 'st',
      two: 'nd',
      few: 'rd',
      other: 'th'
    };
    return `${number}${suffixes[rule] || suffixes.other}`;
  }
}

// Usage example
const plural = new PluralFormatter('en');
const messageTranslations = {
  zero: 'No messages',
  one: '{{count}} message',
  other: '{{count}} messages'
};

console.log(plural.formatPlural(0, messageTranslations)); // "No messages"
console.log(plural.formatPlural(1, messageTranslations)); // "1 message"
console.log(plural.formatPlural(5, messageTranslations)); // "5 messages"

RTL Support

Implement proper right-to-left language support: 

// RTL detection and styling
const RTL_LANGUAGES = ['ar', 'he', 'fa', 'ur'];

class RTLManager {
  constructor() {
    this.isRTL = false;
  }

  setDirection(locale) {
    this.isRTL = RTL_LANGUAGES.includes(locale);
    document.documentElement.dir = this.isRTL ? 'rtl' : 'ltr';
    document.documentElement.classList.toggle('rtl', this.isRTL);
    
    // Update CSS custom properties
    document.documentElement.style.setProperty(
      '--text-align-start', 
      this.isRTL ? 'right' : 'left'
    );
    document.documentElement.style.setProperty(
      '--text-align-end', 
      this.isRTL ? 'left' : 'right'
    );
  }

  adjustFlexDirection(element) {
    if (this.isRTL) {
      const currentDirection = getComputedStyle(element).flexDirection;
      if (currentDirection === 'row') {
        element.style.flexDirection = 'row-reverse';
      }
    }
  }
}

Best Practices

  • Key Organization - Use nested keys for logical grouping and context
  • Fallback Strategy - Always provide fallback translations
  • Context Awareness - Include context in translation keys when needed
  • Performance - Lazy load translations for large applications
  • Cultural Adaptation - Consider cultural differences beyond language
  • Testing - Test with pseudo-locales during development
🎯 Localization Strategy

Successful internationalization requires planning from the start. Design your application architecture to support multiple languages, implement proper text extraction workflows, and consider cultural differences in design and functionality.
