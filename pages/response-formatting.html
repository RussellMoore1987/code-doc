Response Formatting

Implement consistent, well-structured API response formatting to improve client integration, debugging, and overall API usability. This guide covers response standards, error handling, pagination, and content negotiation.

📊 Consistency Matters

Consistent response formatting reduces integration complexity for API consumers, improves debugging efficiency, and creates a professional developer experience that encourages API adoption and reduces support overhead.

Response Formatter

Formatter Implementation

class ResponseFormatter {
  constructor(options = {}) {
    this.options = {
      includeTimestamp: true,
      includeRequestId: true,
      includeProcessingTime: true,
      includeVersion: true,
      version: '1.0',
      ...options
    };
  }

  success(data, meta = {}, links = {}) {
    const response = {
      success: true,
      data,
      meta: this.buildMeta(meta),
    };

    if (Object.keys(links).length > 0) {
      response.links = links;
    }

    return response;
  }

  error(error, statusCode = 500, meta = {}) {
    const response = {
      success: false,
      error: this.formatError(error, statusCode),
      meta: this.buildMeta(meta)
    };

    return response;
  }

  paginated(data, pagination, links = {}, meta = {}) {
    return this.success(data, {
      ...meta,
      pagination: this.formatPagination(pagination)
    }, links);
  }

  buildMeta(customMeta = {}) {
    const meta = { ...customMeta };

    if (this.options.includeTimestamp) {
      meta.timestamp = new Date().toISOString();
    }

    if (this.options.includeRequestId && this.requestId) {
      meta.requestId = this.requestId;
    }

    if (this.options.includeProcessingTime && this.startTime) {
      meta.processingTime = Date.now() - this.startTime;
    }

    if (this.options.includeVersion) {
      meta.version = this.options.version;
    }

    return meta;
  }

  formatError(error, statusCode) {
    if (typeof error === 'string') {
      return {
        code: this.getErrorCode(statusCode),
        message: error
      };
    }

    if (error instanceof Error) {
      return {
        code: error.code || this.getErrorCode(statusCode),
        message: error.message,
        details: error.details || undefined
      };
    }

    return {
      code: error.code || this.getErrorCode(statusCode),
      message: error.message || 'An error occurred',
      details: error.details
    };
  }

  formatPagination(pagination) {
    return {
      page: pagination.page,
      limit: pagination.limit,
      total: pagination.total,
      totalPages: Math.ceil(pagination.total / pagination.limit),
      hasNext: pagination.page * pagination.limit < pagination.total,
      hasPrev: pagination.page > 1
    };
  }

  getErrorCode(statusCode) {
    const codes = {
      400: 'BAD_REQUEST',
      401: 'UNAUTHORIZED',
      403: 'FORBIDDEN',
      404: 'NOT_FOUND',
      409: 'CONFLICT',
      422: 'VALIDATION_ERROR',
      429: 'RATE_LIMIT_EXCEEDED',
      500: 'INTERNAL_ERROR'
    };

    return codes[statusCode] || 'UNKNOWN_ERROR';
  }

  setRequestContext(req) {
    this.requestId = req.id || req.headers['x-request-id'];
    this.startTime = req.startTime || Date.now();
    return this;
  }
}

Express Middleware

// Response formatting middleware
const responseFormatter = (req, res, next) => {
  const formatter = new ResponseFormatter();
  formatter.setRequestContext(req);

  // Add helper methods to response object
  res.success = (data, meta, links) => {
    const response = formatter.success(data, meta, links);
    return res.json(response);
  };

  res.error = (error, statusCode = 500, meta) => {
    const response = formatter.error(error, statusCode, meta);
    return res.status(statusCode).json(response);
  };

  res.paginated = (data, pagination, links, meta) => {
    const response = formatter.paginated(data, pagination, links, meta);
    return res.json(response);
  };

  res.created = (data, meta) => {
    const response = formatter.success(data, meta);
    return res.status(201).json(response);
  };

  res.noContent = () => {
    return res.status(204).send();
  };

  next();
};

Content Negotiation

Format Negotiation

class ContentNegotiator {
  constructor() {
    this.formatters = new Map([
      ['application/json', this.formatJSON],
      ['application/xml', this.formatXML],
      ['text/csv', this.formatCSV],
      ['text/plain', this.formatText]
    ]);
  }

  negotiate(req, data) {
    const acceptHeader = req.get('Accept') || 'application/json';
    const preferredType = this.parseAcceptHeader(acceptHeader);
    
    const formatter = this.formatters.get(preferredType) || 
                     this.formatters.get('application/json');
    
    return {
      contentType: preferredType,
      data: formatter(data)
    };
  }

  parseAcceptHeader(acceptHeader) {
    const types = acceptHeader.split(',').map(type => {
      const [mediaType, ...params] = type.trim().split(';');
      const quality = params.find(p => p.trim().startsWith('q='));
      const q = quality ? parseFloat(quality.split('=')[1]) : 1.0;
      
      return { mediaType: mediaType.trim(), quality: q };
    });

    // Sort by quality score
    types.sort((a, b) => b.quality - a.quality);
    
    // Return the first supported type
    for (const type of types) {
      if (this.formatters.has(type.mediaType)) {
        return type.mediaType;
      }
    }
    
    return 'application/json';
  }

  formatJSON(data) {
    return JSON.stringify(data, null, 2);
  }

  formatXML(data) {
    const toXML = (obj, rootName = 'response') => {
      let xml = `<${rootName}>`;
      
      for (const [key, value] of Object.entries(obj)) {
        if (Array.isArray(value)) {
          xml += `<${key}>`;
          value.forEach(item => {
            xml += toXML(item, 'item');
          });
          xml += ``;
        } else if (typeof value === 'object' && value !== null) {
          xml += toXML(value, key);
        } else {
          xml += `<${key}>${this.escapeXML(value)}`;
        }
      }
      
      xml += ``;
      return xml;
    };
    
    return '\n' + toXML(data);
  }

  formatCSV(data) {
    if (!Array.isArray(data.data)) {
      throw new Error('CSV format requires array data');
    }

    const items = data.data;
    if (items.length === 0) return '';

    const headers = Object.keys(items[0]);
    const csvRows = [headers.join(',')];

    items.forEach(item => {
      const values = headers.map(header => {
        const value = item[header];
        return typeof value === 'string' && value.includes(',') 
          ? `"${value}"` 
          : value;
      });
      csvRows.push(values.join(','));
    });

    return csvRows.join('\n');
  }

  formatText(data) {
    return JSON.stringify(data, null, 2);
  }

  escapeXML(value) {
    if (typeof value !== 'string') return value;
    
    return value
      .replace(/&/g, '&')
      .replace(//g, '>')
      .replace(/"/g, '"')
      .replace(/'/g, ''');
  }
}

Response Compression

Compression Middleware

const compression = require('compression');
const zlib = require('zlib');

// Smart compression based on content type and size
const smartCompression = compression({
  // Only compress responses larger than 1KB
  threshold: 1024,
  
  // Compression level (1-9, higher = better compression but slower)
  level: 6,
  
  // Compress these MIME types
  filter: (req, res) => {
    if (req.headers['x-no-compression']) {
      return false;
    }
    
    const contentType = res.get('Content-Type');
    const compressibleTypes = [
      'application/json',
      'application/xml',
      'text/plain',
      'text/html',
      'text/css',
      'application/javascript'
    ];
    
    return compressibleTypes.some(type => 
      contentType && contentType.includes(type)
    );
  },

  // Use gzip for most clients, brotli for modern browsers
  strategy: (req, res) => {
    const acceptEncoding = req.get('Accept-Encoding') || '';
    
    if (acceptEncoding.includes('br')) {
      return zlib.constants.BROTLI_ENCODE;
    } else if (acceptEncoding.includes('gzip')) {
      return zlib.constants.GZIP;
    }
    
    return null; // No compression
  }
});

// Response size monitoring
const responseSizeMonitor = (req, res, next) => {
  const originalSend = res.send;
  const originalJson = res.json;
  
  res.send = function(body) {
    if (body) {
      res.set('X-Response-Size', Buffer.byteLength(body));
    }
    return originalSend.call(this, body);
  };
  
  res.json = function(obj) {
    const jsonString = JSON.stringify(obj);
    res.set('X-Response-Size', Buffer.byteLength(jsonString));
    return originalJson.call(this, obj);
  };
  
  next();
};

Caching Headers

Cache Control

class CacheHeaderManager {
  static setCacheHeaders(res, options = {}) {
    const {
      maxAge = 0,
      private = false,
      noCache = false,
      noStore = false,
      mustRevalidate = false,
      staleWhileRevalidate = 0,
      etag = null
    } = options;

    let cacheControl = [];

    if (noStore) {
      cacheControl.push('no-store');
    } else if (noCache) {
      cacheControl.push('no-cache');
    } else {
      cacheControl.push(private ? 'private' : 'public');
      cacheControl.push(`max-age=${maxAge}`);
      
      if (staleWhileRevalidate > 0) {
        cacheControl.push(`stale-while-revalidate=${staleWhileRevalidate}`);
      }
    }

    if (mustRevalidate) {
      cacheControl.push('must-revalidate');
    }

    res.set('Cache-Control', cacheControl.join(', '));

    if (etag) {
      res.set('ETag', `"${etag}"`);
    }

    // Set Vary header for content negotiation
    res.set('Vary', 'Accept, Accept-Encoding');
  }

  static cacheMiddleware(options) {
    return (req, res, next) => {
      // Set cache headers based on route or resource type
      const cacheConfig = this.getCacheConfig(req.path, options);
      this.setCacheHeaders(res, cacheConfig);
      
      next();
    };
  }

  static getCacheConfig(path, options) {
    const configs = {
      '/api/users': { maxAge: 300, private: true }, // 5 minutes
      '/api/posts': { maxAge: 600 }, // 10 minutes
      '/api/config': { maxAge: 3600 }, // 1 hour
      '/api/auth': { noCache: true }
    };

    return configs[path] || options.default || { maxAge: 0 };
  }
}

Best Practices

  • Consistency - Use the same response format across all endpoints
  • Meaningful Errors - Provide clear, actionable error messages
  • Proper Status Codes - Use appropriate HTTP status codes
  • Pagination - Always paginate large result sets
  • Metadata - Include relevant metadata like timestamps and request IDs
  • Compression - Compress responses to reduce bandwidth usage
  • Caching - Set appropriate cache headers for performance
🔄 Response Evolution

Design your response format to be extensible. Use versioning strategies that allow you to evolve your API without breaking existing clients. Consider backwards compatibility and provide migration paths for major changes.
