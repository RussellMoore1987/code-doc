Session Management

Implement secure, scalable session management to maintain user state across requests. This guide covers session storage options, security considerations, and performance optimization techniques.

🔐 Security Focus

Sessions contain sensitive user data and authentication state. Proper session management is critical for application security, preventing session hijacking, fixation attacks, and unauthorized access.

Session Basics

Sessions allow servers to maintain user state across multiple HTTP requests. They work by storing a session identifier on the client and session data on the server.

  });

  res.json({ message: 'Login successful' });
});

Session Middleware

const sessionMiddleware = async (req, res, next) => {
  const sessionId = req.cookies.sessionId;
  
  if (!sessionId) {
    req.session = null;
    return next();
  }

  try {
    const session = await sessionStore.get(sessionId);
    
    if (!session) {
      res.clearCookie('sessionId');
      req.session = null;
      return next();
    }

    // Check session expiry
    if (session.expiresAt && session.expiresAt < new Date()) {
      await sessionStore.destroy(sessionId);
      res.clearCookie('sessionId');
      req.session = null;
      return next();
    }

    // Update last activity
    session.lastActivity = new Date();
    await sessionStore.update(sessionId, session);
    
    req.session = session;
    req.sessionId = sessionId;
    
  } catch (error) {
    console.error('Session middleware error:', error);
    req.session = null;
  }
  
  next();
};

Storage Options

Memory Store

class MemorySessionStore {
  constructor() {
    this.sessions = new Map();
    this.startCleanupInterval();
  }

  async create(sessionId, data) {
    this.sessions.set(sessionId, {
      ...data,
      createdAt: new Date(),
      expiresAt: new Date(Date.now() + (24 * 60 * 60 * 1000))
    });
  }

  async get(sessionId) {
    return this.sessions.get(sessionId) || null;
  }

  async update(sessionId, data) {
    if (this.sessions.has(sessionId)) {
      this.sessions.set(sessionId, data);
    }
  }

  async destroy(sessionId) {
    this.sessions.delete(sessionId);
  }

  startCleanupInterval() {
    setInterval(() => {
      const now = new Date();
      for (const [sessionId, session] of this.sessions) {
        if (session.expiresAt && session.expiresAt < now) {
          this.sessions.delete(sessionId);
        }
      }
    }, 15 * 60 * 1000); // Every 15 minutes
  }
}

Redis Store

const Redis = require('redis');

class RedisSessionStore {
  constructor(redisConfig) {
    this.client = Redis.createClient(redisConfig);
    this.prefix = 'session:';
    this.defaultTTL = 24 * 60 * 60; // 24 hours
  }

  async create(sessionId, data) {
    const key = this.prefix + sessionId;
    const serialized = JSON.stringify(data);
    await this.client.setex(key, this.defaultTTL, serialized);
  }

  async get(sessionId) {
    const key = this.prefix + sessionId;
    const data = await this.client.get(key);
    return data ? JSON.parse(data) : null;
  }

  async update(sessionId, data) {
    const key = this.prefix + sessionId;
    const ttl = await this.client.ttl(key);
    const serialized = JSON.stringify(data);
    
    if (ttl > 0) {
      await this.client.setex(key, ttl, serialized);
    } else {
      await this.client.setex(key, this.defaultTTL, serialized);
    }
  }

  async destroy(sessionId) {
    const key = this.prefix + sessionId;
    await this.client.del(key);
  }

  async extend(sessionId, additionalTime = this.defaultTTL) {
    const key = this.prefix + sessionId;
    await this.client.expire(key, additionalTime);
  }
}

Session Regeneration

async function regenerateSession(req, res) {
  const oldSessionId = req.sessionId;
  const oldSession = req.session;
  
  // Generate new session ID
  const newSessionId = generateSecureId();
  
  // Copy session data to new session
  const newSession = {
    ...oldSession,
    id: newSessionId,
    regeneratedAt: new Date()
  };
  
  // Create new session
  await sessionStore.create(newSessionId, newSession);
  
  // Remove old session
  if (oldSessionId) {
    await sessionStore.destroy(oldSessionId);
  }
  
  // Update cookie
  res.cookie('sessionId', newSessionId, {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    sameSite: 'strict',
    maxAge: 24 * 60 * 60 * 1000
  });
  
  req.sessionId = newSessionId;
  req.session = newSession;
}

Concurrent Session Control

class ConcurrentSessionManager {
  constructor(sessionStore, maxSessions = 3) {
    this.sessionStore = sessionStore;
    this.maxSessions = maxSessions;
    this.userSessions = new Map();
  }

  async createSession(userId, sessionData) {
    const sessionId = generateSecureId();
    
    // Track user sessions
    if (!this.userSessions.has(userId)) {
      this.userSessions.set(userId, new Set());
    }
    
    const sessions = this.userSessions.get(userId);
    
    // Enforce session limit
    if (sessions.size >= this.maxSessions) {
      const oldestSession = sessions.values().next().value;
      await this.destroySession(oldestSession);
      sessions.delete(oldestSession);
    }
    
    sessions.add(sessionId);
    
    await this.sessionStore.create(sessionId, {
      ...sessionData,
      userId,
      sessionId
    });
    
    return sessionId;
  }

  async destroySession(sessionId) {
    const session = await this.sessionStore.get(sessionId);
    if (session) {
      const sessions = this.userSessions.get(session.userId);
      if (sessions) {
        sessions.delete(sessionId);
        if (sessions.size === 0) {
          this.userSessions.delete(session.userId);
        }
      }
    }
    
    await this.sessionStore.destroy(sessionId);
  }

  async getUserSessions(userId) {
    const sessionIds = this.userSessions.get(userId) || new Set();
    const sessions = [];
    
    for (const sessionId of sessionIds) {
      const session = await this.sessionStore.get(sessionId);
      if (session) {
        sessions.push(session);
      } else {
        // Clean up stale session ID
        sessionIds.delete(sessionId);
      }
    }
    
    return sessions;
  }
}

Session Clustering

// Consistent hashing for session distribution
class SessionCluster {
  constructor(servers) {
    this.servers = servers;
    this.ring = new Map();
    this.buildHashRing();
  }

  buildHashRing() {
    this.servers.forEach(server => {
      for (let i = 0; i < 160; i++) {
        const hash = this.hash(`${server.host}:${server.port}:${i}`);
        this.ring.set(hash, server);
      }
    });
  }

  getServerForSession(sessionId) {
    const hash = this.hash(sessionId);
    const keys = Array.from(this.ring.keys()).sort((a, b) => a - b);
    
    for (const key of keys) {
      if (key >= hash) {
        return this.ring.get(key);
      }
    }
    
    return this.ring.get(keys[0]);
  }

  hash(str) {
    let hash = 0;
    for (let i = 0; i < str.length; i++) {
      hash = ((hash << 5) - hash + str.charCodeAt(i)) & 0xffffffff;
    }
    return Math.abs(hash);
  }
}

Best Practices

  • Secure Cookies - Always use httpOnly, secure, and sameSite flags
  • Session Expiration - Implement both idle timeout and absolute timeout
  • Regenerate on Privilege Change - Generate new session ID after authentication
  • IP and User-Agent Validation - Detect suspicious session usage
  • Proper Cleanup - Implement automated session garbage collection
  • Audit Logging - Log session creation, destruction, and suspicious activity
🔄 Session Lifecycle

Proper session lifecycle management is crucial for both security and performance. Always implement session expiration, cleanup mechanisms, and regeneration on sensitive operations to maintain a secure application state.
